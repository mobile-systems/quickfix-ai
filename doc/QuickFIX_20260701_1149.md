void QuoteApp::fromApp(const FIX::Message& msg, const FIX::SessionID& sid) {
    try {
        if (msg.getHeader().getField(FIX::FIELD::MsgType) == FIX::MsgType_MarketDataSnapshotFullRefresh) {
            onMessage(static_cast<const FIX::MarketDataSnapshotFullRefresh&>(msg), sid);
        } else if (msg.getHeader().getField(FIX::FIELD::MsgType) == FIX::MsgType_MarketDataIncrementalRefresh) {
            onMessage(static_cast<const FIX::MarketDataIncrementalRefresh&>(msg), sid);
        } else {
            std::cout << "Unhandled message type: "
                      << msg.getHeader().getField(FIX::FIELD::MsgType) << "\n";

        std::cerr << "Field not found: " << e.what() << "\n";

void QuoteApp::onMessage(const FIX::MarketDataSnapshotFullRefresh& msg,
                         const FIX::SessionID&) {
    std::string symbol;
    double bid = 0.0, ask = 0.0;
    if (msg.getField(FIX::Field<FIX::Symbol>(FIX::FIELD::Symbol), symbol)) {
        // –ü–ĺ–Ľ—É—á–į–Ķ–ľ –Ņ–Ķ—Ä–≤—č–Ļ —ć–Ľ–Ķ–ľ–Ķ–Ĺ—ā MDEntry
        const FIX::MDEntries entries = msg.getGroup(1, FIX::MDEntries());
        for (int i = 1; i <= entries.size(); ++i) {
            FIX::MDEntryType type;
            FIX::MDEntryPx px;
            entries.getField(i, type);
            entries.getField(i, px);
            if (type.getValue() == '0') bid = px.getValue(); // BID
            if (type.getValue() == '1') ask = px.getValue(); // ASK
        }
        if (!symbol.empty() && (bid > 0 || ask > 0)) {
            db_.insertQuote(symbol, bid, ask);
            std::cout << "Stored EURUSD quote: BID=" << bid << " ASK=" << ask << "\n";
        }
    }
}
