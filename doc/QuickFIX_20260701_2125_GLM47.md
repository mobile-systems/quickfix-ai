```cpp
#ifndef TRADERAPP_H
#define TRADERAPP_H

#include <quickfix/Session.h>
#include <quickfix/Message.h>
#include <quickfix/fix42/NewOrderSingle.h>
#include <quickfix/fix42/ExecutionReport.h>
#include <quickfix/fix42/MarketDataSnapshotFullRefresh.h>
#include <quickfix/fix42/MarketDataIncrementalRefresh.h>
#include <quickfix/fix42/Heartbeat.h>
#include <pqxx/pqxx>
#include <queue>
#include <mutex>

// Простая обертка для асинхронной записи в БД
class DatabaseWorker {
public:
    DatabaseWorker(std::string connStr) : connection_(connStr) {
        worker_thread_ = std::thread(&DatabaseWorker::processQueue, this);
    }

    ~DatabaseWorker() {
        stop_ = true;
        if (worker_thread_.joinable()) worker_thread_.join();
    }

    // Метод для вставки данных из потока FIX (HFT подход: не блокировать callbacks)
    void enqueue(std::string sql) {
        std::lock_guard<std::mutex> lock(queue_mutex_);
        db_queue_.push(sql);
    }

private:
    pqxx::connection connection_;
    std::queue<std::string> db_queue_;
    std::mutex queue_mutex_;
    std::thread worker_thread_;
    bool stop_ = false;

    void processQueue() {
        while (!stop_) {
            std::string sql;
            {
                std::lock_guard<std::mutex> lock(queue_mutex_);
                if (db_queue_.empty()) {
                    std::this_thread::sleep_for(std::chrono::milliseconds(10));
                    continue;
                }
                sql = std::move(db_queue_.front());
                db_queue_.pop();
            }
            try {
                if (!connection_.is_open()) {
                    connection_.connect();
                }
                pqxx::work txn(connection_);
                txn.exec(sql);
                txn.commit();
            } catch (const std::exception& e) {
                // В HFT логгирование ошибок БД критично, но здесь для упрощения просто вывод
                std::cerr << "[DB ERROR] " << e.what() << std::endl;
            }
        }
    }
};

class TraderApp : public QuickFIX::Application {
public:
    TraderApp(DatabaseWorker& dbWorker) : dbWorker_(dbWorker) {}

    void onCreate(const QuickFIX::SessionID& session) override {}
    void onLogon(const QuickFIX::SessionID& session) override {
        std::string msg = "INSERT INTO sessions (session_id, status, ts) VALUES ('" +
                          session.toString() + "', 'LOGON', NOW());";
        dbWorker_.enqueue(msg);
        std::cout << "Session " << session.toString() << " logged on." << std::endl;
    }
    void onLogout(const QuickFIX::SessionID& session) override {
        std::string msg = "INSERT INTO sessions (session_id, status, ts) VALUES ('" +
                          session.toString() + "', 'LOGOUT', NOW());";
        dbWorker_.enqueue(msg);
        std::cout << "Session " << session.toString() << " logged off." << std::endl;
    }
    void toAdmin(const QuickFIX::Message& message, const QuickFIX::SessionID& session) override {}
    void toApp(const QuickFIX::Message& message, const QuickFIX::SessionID& session) override {}

    // Обработка сообщений
    void onMessage(const QuickFIX::Message& message, const QuickFIX::SessionID& session) override {
        try {
            const std::string msgType = message.getHeader().getFieldString(QuickFIX::MsgType);

            // Обработка Market Data (Snapshot)
            if (msgType == "W") {
                const QuickFIX::Message& mdMsg = static_cast<const QuickFIX::Message&>(message);
                const QuickFIX::FieldMap& body = mdMsg.getBody();

                // Пример простого извлечения (реально нужен парсинг XML/FlatFix)
                // Здесь мы упростим: будем записывать факт получения котировки в БД
                // для примера HFT, так как полный парсинг XML FIX в C++ без сторонних библиотек сложен.
                // Для полноценного HFT используют генераторы кода QuickFIX++.
            }

            // Обработка Execution Report (8)
            if (msgType == "8") {
                std::string symbol = "", side = "", qty = "", price = "";
                const QuickFIX::Message& execMsg = static_cast<const QuickFIX::Message&>(message);

                // Извлечение полей (упрощенно для примера)
                try {
                    symbol = execMsg.getFieldString(QuickFIX::Symbol);
                    side = execMsg.getFieldString(QuickFIX::Side);
                    qty = execMsg.getFieldString(QuickFIX::OrderQty);
                    price = execMsg.getFieldString(QuickFIX::Price);

                    // Асинхронная запись сделки (не блокирует FIX поток)
                    std::string sql = "INSERT INTO trades (symbol, side, qty, price, timestamp) VALUES ('" +
                                      symbol + "', '" + side + "', '" + qty + "', " + price + ", NOW());";
                    dbWorker_.enqueue(sql);

                    std::cout << "TRADE: " << symbol << " @ " << price << " (" << side << " x " << qty << ")" << std::endl;
                } catch (...) {}
            }
        } catch (const std::exception& e) {
            std::cerr << "Error processing message: " << e.what() << std::endl;
        }
    }

private:
    DatabaseWorker& dbWorker_;
};

#endif
```

#### `main.cpp`

```cpp
#include <iostream>
#include <quickfix/SessionSettings.h>
#include <quickfix/FileStore.h>
#include <quickfix/SocketInitiator.h>
#include <quickfix/ScreenLog.h>
#include <quickfix/ThreadedSocketConnector.h>
#include <quickfix/fix42/Message.h>
#include <pqxx/pqxx>
#include "TraderApp.h"
#include <memory>

int main() {
    std::string db_conn_str = "postgresql://trader:trader@postgres:5432/hft_db";

    // Подключение к БД для создания таблиц при старте (опционально)
    try {
        pqxx::connection conn(db_conn_str);
        if (conn.is_open()) {
            std::cout << "Connected to DB." << std::endl;
            pqxx::work txn(conn);

            // Создание таблиц
            txn.exec(
                "CREATE TABLE IF NOT EXISTS trades ("
                "id SERIAL PRIMARY KEY,"
                "symbol VARCHAR(10),"
                "side VARCHAR(4),"
                "qty NUMERIC(20, 6),"
                "price NUMERIC(20, 8),"
                "timestamp TIMESTAMP DEFAULT NOW()"
                ");"
            );
            txn.exec(
                "CREATE TABLE IF NOT EXISTS sessions ("
                "id SERIAL PRIMARY KEY,"
                "session_id VARCHAR(255),"
                "status VARCHAR(10),"
                "ts TIMESTAMP DEFAULT NOW()"
                ");"
            );
            txn.commit();
        }
    } catch (const std::exception& e) {
        std::cerr << "DB Connection failed: " << e.what() << std::endl;
        return 1;
    }

    DatabaseWorker dbWorker(db_conn_str);

    try {
        // Загрузка настроек (trader.cfg должен быть рядом)
        QuickFIX::SessionSettings settings("trader.cfg");

        // Создание приложения
        std::unique_ptr<QuickFIX::Application> app(new TraderApp(dbWorker));

        // Использование потокового сокет-инициатора для многопоточности
        QuickFIX::SocketInitiator initiator(*app, settings, new QuickFIX::ThreadedSocketConnector());

        initiator.start();

        std::cout << "HFT Trader started. Listening for FIX messages..." << std::endl;

        // Цикл работы
        while (true) {
            std::this_thread::sleep_for(std::chrono::seconds(1));
        }

        initiator.stop();
    } catch (QuickFIX::ConfigError& e) {
        std::cerr << "Config Error: " << e.what() << std::endl;
        return 1;
    }

    return 0;
}
```

### 1. Новая структура данных (`DatabaseWorker.h`)

Мы создадим структуру для данных о сделках и сессиях и будем передавать их в очередь. Это решает проблему разных типов аргументов.

```cpp
#ifndef DATABASEWORKER_H
#define DATABASEWORKER_H

#include <pqxx/pqxx>
#include <pqxx/sql>
#include <queue>
#include <mutex>
#include <variant>
#include <thread>
#include <chrono>

// Структуры для хранения данных разных типов в одном контейнере
struct TradeData {
    std::string symbol;
    std::string side;
    double qty;
    double price;
};

struct SessionData {
    std::string session_id;
    std::string status;
};

class DatabaseWorker {
public:
    DatabaseWorker(std::string connStr) : connection_(connStr) {
        worker_thread_ = std::thread(&DatabaseWorker::processQueue, this);
    }

    ~DatabaseWorker() {
        stop_ = true;
        if (worker_thread_.joinable()) worker_thread_.join();
    }

    // Методы для отправки данных (принимают значения, а не SQL)
    void enqueueTrade(const std::string& sym, const std::string& side, double qty, double price) {
        std::lock_guard<std::mutex> lock(queue_mutex_);
        db_queue_.emplace(TradeData{sym, side, qty, price});
    }

    void enqueueSession(const std::string& sid, const std::string& status) {
        std::lock_guard<std::mutex> lock(queue_mutex_);
        db_queue_.emplace(SessionData{sid, status});
    }

    // Инициализация SQL (вызывается один раз в main)
    void prepare(pqxx::connection &conn) {
        // Определяем SQL запросы с плейсхолдерами ($1, $2, ...)
        // Это безопасно и эффективно
        insert_trade_sql = pqxx::sql::query<
            "INSERT INTO trades (symbol, side, qty, price, timestamp) VALUES ($1, $2, $3, $4, DEFAULT)"
        >{};

        insert_session_sql = pqxx::sql::query<
            "INSERT INTO sessions (session_id, status, ts) VALUES ($1, $2, DEFAULT)"
        >{};
    }

private:
    pqxx::connection connection_;
    std::queue<std::variant<TradeData, SessionData>> db_queue_;
    std::mutex queue_mutex_;
    std::thread worker_thread_;
    bool stop_ = false;

    // Объекты SQL запросов
    pqxx::sql::query<> insert_trade_sql;
    pqxx::sql::query<> insert_session_sql;

    void processQueue() {
        // Проверка соединения при старте
        if (!connection_.is_open()) connection_.connect();

        while (!stop_) {
            std::variant<TradeData, SessionData> item;
            {
                std::lock_guard<std::mutex> lock(queue_mutex_);
                if (db_queue_.empty()) {
                    std::this_thread::sleep_for(std::chrono::milliseconds(1));
                    continue;
                }
                item = std::move(db_queue_.front());
                db_queue_.pop();
            }

            try {
                // Начинаем транзакцию для каждого блока данных
                pqxx::work txn(connection_);

                if (std::holds_alternative<TradeData>(item)) {
                    auto t = std::get<TradeData>(item);
                    // Выполняем параметризованный запрос
                    // $1 - symbol, $2 - side, $3 - qty, $4 - price
                    txn.exec_prepared(insert_trade, t.symbol, t.side, t.qty, t.price);
                }
                else if (std::holds_alternative<SessionData>(item)) {
                    auto s = std::get<SessionData>(item);
                    // Выполняем параметризованный запрос
                    txn.exec_prepared(insert_session, s.session_id, s.status);
                }

                txn.commit(); // Фиксируем изменения
            } catch (const std::exception& e) {
                // Логирование ошибок транзакции (опционально)
                // Для HFT важна скорость, поэтому ошибки в лог, а не прерывание потока
            }
        }
    }
};
```
