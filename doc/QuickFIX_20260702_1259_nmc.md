{
    std::cout << "[FixApp] Логин в сессию " << std::endl;
    // Сразу запросим начальные котировки после логина.
    // Схема запроса MarketDataRequest (MDR) определяется ниже.
    Session *session = Session::lookupSession("HFT_FIX");
    if (!session) return;
    Message mdReq;
    FieldMap& fmap = mdReq.getFieldMap();
    // Начинаем с поля 35 = d (MarketDataRequest)
    mdReq.set(Field<35>(FIELD::MsgType::MARKET_DATA_REQUEST));
    mdReq.set(Field<34>(42)); // MsgSeqNum – будет установлено автоматически
    // Упростим: будем запрашивать все 12 инструментов в одном запросе
    // (реальный HFT будет генерировать MDR для каждого инструмента с высокой частотой)
    // Мы используем быстрый ручной запрос: 55=instrument, 262=reqId, 263=2 (SNAPSHOT), 264=0 (уровни)
    for (const auto &ins : instruments_)
    {
        Field<55> fld55(ins);
        mdReq.setField(fld55);
    }
    mdReq.set(Field<262>("SNAPSHOT_001"));
    mdReq.set(Field<263>(2));
    mdReq.set(Field<264>(0));
    // Отправляем
    session->sendToTarget(mdReq, SessionID("HFT_FIX", "", ""));
}

    q.timestamp = nowMs();
    // FIX-поля: 55=Symbol (инструмент), 6=AskPx, 4=BidPx, 1=LastPx

void DbManager::createTables()
{
    // Подключаемся через дефолтный коннект, если еще не сделали это
    if (!conn_)
        throw std::runtime_error("DbManager не подключен");

    work txn(*conn_);
    txn.exec(R"(
        CREATE TABLE IF NOT EXISTS quotes (
            id SERIAL PRIMARY KEY,
            instrument TEXT NOT NULL,
            bid NUMERIC,
            ask NUMERIC,
            last NUMERIC,
            ts BIGINT NOT NULL

        CREATE TABLE IF NOT EXISTS trades (
            id SERIAL PRIMARY KEY,

### 3.8 `src/PeriodicFetcher.cpp`

```cpp
#include "PeriodicFetcher.h"
#include <iostream>
#include <atomic>

PeriodicFetcher::PeriodicFetcher(FixApplication &app) : app_(std::addressof(app))
{
    running_.store(false);
}

void PeriodicFetcher::start(uint64_t interval_ms)
{
    if (running_) return;
    running_.store(true);
    worker_ = std::thread(&PeriodicFetcher::run, this, interval_ms);
}

void PeriodicFetcher::stop()
{
    if (!running_) return;
    running_.store(false);
    if (worker_.joinable())
        worker_.join();
}

void PeriodicFetcher::run(uint64_t interval_ms)
{
    while (running_)
    {
        app_->persistSnapshot(); // сохраняем всё, что у нас есть
        std::this_thread::sleep_for(std::chrono::milliseconds(interval_ms));
    }
}
```

Повторите шаги для таблицы `trades` (измените запрос на `SELECT * FROM trades ORDER BY trade_time DESC LIMIT 1000;`).

### 7.2 Создать панели визуализации

#### Панель 1 – Последние котировки (таблица)

* **Добавить панель** → **Таблица**
  * Источник данных = **PostgreSQL (quotes)**
  * **Запрос** = `SELECT instrument, bid, ask, last, ts FROM quotes ORDER BY ts DESC LIMIT 100`
  * **Массив меток** = `{ "instrument": "${instrument}" }` (если вам нужен динамический фильтр)
  * Настройте столбцы, сортировку по `ts`, автопрокрутку.

#### Панель 2 – Распределение сделок (диаграмма)

* **Добавить панель** → **Диаграмма`
  * Источник данных = **PostgreSQL (trades)**
  * **Запрос** = `SELECT side, COUNT(*) AS cnt FROM trades GROUP BY side`
  * **Тип** = **Pie` (или **Bar**)
  * **Этикетки** = `side`
  * **Значение** = `cnt`

## 8⃣ Как это работает – краткое объяснение

| Компонент | Роль |
|-----------|------|
| **QuickFix++** | Парсит FIX-сообщения (`MarketDataSnapshotFullRefresh`, `ExecutionReport`) и поддерживает сессию. |
| **FixApplication** | Наследуется от `Application` и `MessageCracker`. Накапливает котировки и сделки в памяти, периодически сохраняет их в PostgreSQL. |
| **PeriodicFetcher** | Boost.Asio не используется напрямую здесь, но `std::thread` выполняет `persistSnapshot` каждые `interval_ms` (настраиваемый, например 2 секунды). Это обеспечивает **не-блокирующий поток** для обработки FIX-сообщений, которые приходят асинхронно. |
| **DbManager** | Обертка над libpq/pqxx. Простые запросы `INSERT`/`SELECT` для хранения и последующего извлечения данных для Grafana. |
| **CMake** | Собирает проект с зависимостями Boost.Asio, libpqxx, QuickFix++. |
| **Docker + Compose** | Изолирует приложение, PostgreSQL и Grafana; легко развертывать в продакшене или тестовой среде. |
| **Grafana** | Визуализирует данные котировок/сделок, предоставляя панель мониторинга в режиме реального времени. |
