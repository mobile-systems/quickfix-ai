#include "db_manager.hpp"
#include <stdexcept>

DbManager::DbManager(const std::string& conn_str) : m_conn_str(conn_str) {}

void DbManager::insertQuotes(const std::vector<Quote>& quotes) {
    pqxx::connection conn(m_conn_str);
    pqxx::work txn(conn);
    std::string sql = R"(
        INSERT INTO quotes (symbol, bid, ask, last, timestamp)
        VALUES ($1, $2, $3, $4, to_timestamp($5::bigint/1000000000))
    )";
    for (const auto& q : quotes) {
        txn.exec_prepared("q_ins", q.symbol, q.bid, q.ask, q.last, q.timestamp_ns);
    }
    txn.commit();
}

void DbManager::insertTrades(const std::vector<Trade>& trades) {
    pqxx::connection conn(m_conn_str);
    pqxx::work txn(conn);
    std::string sql = R"(
        INSERT INTO trades (order_id, symbol, side, ord_type, price, qty, exec_type, cum_qty, leaves_qty, timestamp)
        VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9, to_timestamp($10::bigint/1000000000))
    )";
    for (const auto& t : trades) {
        txn.exec_prepared("t_ins", t.order_id, t.symbol, t.side, t.ord_type, t.price, t.qty, t.exec_type, t.cum_qty, t.leaves_qty, t.timestamp_ns);
    }
    txn.commit();
}

void DbManager::insertBalances(const std::vector<Balance>& bals) {
    pqxx::connection conn(m_conn_str);
    pqxx::work txn(conn);
    std::string sql = R"(
        INSERT INTO balances (account, cash, equity, timestamp)
        VALUES ($1, $2, $3, to_timestamp($4::bigint/1000000000))
    )";
    for (const auto& b : bals) {
        txn.exec_prepared("b_ins", b.account, b.cash, b.equity, b.timestamp_ns);

#pragma once
#include <array>
#include <atomic>
#include <cstddef>
#include <cstdint>
#include <type_traits>

template<typename T, std::size_t Capacity>
requires (Capacity > 0) && ((Capacity & (Capacity - 1)) == 0) // power of 2
struct alignas(64) RingBuffer {
    static constexpr std::size_t Mask = Capacity - 1;
    static constexpr std::size_t CachePad = 64 / sizeof(T);

    alignas(64) std::atomic<std::size_t> tail{0};
    alignas(64) std::atomic<std::size_t> head{0};
    alignas(64) T buffer[Capacity];
    char pad[CachePad * sizeof(T) - sizeof(std::atomic<std::size_t>) * 2 - Capacity * sizeof(T)];

    [[nodiscard]] bool push(const T& val) {
        std::size_t t = tail.load(std::memory_order_relaxed);
        std::size_t next = (t + 1) & Mask;
        if (next == head.load(std::memory_order_acquire)) return false;
        buffer[t] = val;
        tail.store(next, std::memory_order_release);
        return true;
    }

    [[nodiscard]] bool pop(T& val) {
        std::size_t h = head.load(std::memory_order_relaxed);
        if (h == tail.load(std::memory_order_acquire)) return false;
        val = buffer[h];
        head.store((h + 1) & Mask, std::memory_order_release);
        return true;
    }
};

struct DbParamBatch {

    const char* values[8]; // max 8 –Ņ–į—Ä–į–ľ–Ķ—ā—Ä–ĺ–≤ (trade/quote/balance)

class DbAsyncWriter {

    std::thread worker_;
    std::atomic<bool> running_{true};

    std::queue<std::vector<DbParamBatch>> batch_queue_;
    std::mutex mtx_;
    std::condition_variable cv_;

    void init_connection() {
        std::string conn_str = "host=postgres port=5432 user=trader password=secret dbname=hft_db "
                               "connect_timeout=5 keepalives=1 keepalives_idle=10 keepalives_interval=5 keepalives_count=5";

        if (PQstatus(conn_) != CONNECTION_OK) {

                                 "INSERT INTO balances (account, cash, equity) VALUES ($1, $2, $3)";

        PQexec(conn_, stmt_quote);
        PQexec(conn_, stmt_bal);
    }

    void flush_batch() {

        if (batch_queue_.empty()) return;

        auto batch = std::move(batch_queue_.front());
        batch_queue_.pop();

        // –í—č–Ņ–ĺ–Ľ–Ĺ—Ź–Ķ–ľ –Ņ–į—Ä–į–ľ–Ķ—ā—Ä—č

            PGresult* res = nullptr;

            // –í—č–Ī–ł—Ä–į–Ķ–ľ statement –Ņ–ĺ –ļ–ĺ–Ľ-–≤—É –Ņ–į—Ä–į–ľ–Ķ—ā—Ä–ĺ–≤

                                                  p.values, p.lengths, p.formats, 0);
            else if (p.count == 4) res = PQexecParams(conn_, "ins_quote", p.count, nullptr,
                                                       p.values, p.lengths, p.formats, 0);
            else if (p.count == 3) res = PQexecParams(conn_, "ins_bal", p.count, nullptr,
                                                       p.values, p.lengths, p.formats, 0);
            else continue;

                std::cerr << "[DB] Error: " << PQresultErrorMessage(res) << "\n";
            }
            PQclear(res);
        }
    }

    void worker_func() {
        init_connection();

        while (running_.load(std::memory_order_relaxed)) {
            std::unique_lock<std::mutex> lock(mtx_);
            cv_.wait_for(lock, std::chrono::milliseconds(2),
                         [this]{ return !batch_queue_.empty() || !running_.load(); });

            flush_batch();
        }

        PQfinish(conn_);
    }

public:
    DbAsyncWriter() {
        worker_ = std::thread(&DbAsyncWriter::worker_func, this);
    }
    ~DbAsyncWriter() {
        running_.store(false, std::memory_order_release);
        cv_.notify_one();
        worker_.join();
    }

                    double price, double qty) {
        static thread_local char buf[256];
        static thread_local int pos = 0;

        // –§–ĺ—Ä–ľ–į—ā–ł—Ä–ĺ–≤–į–Ĺ–ł–Ķ –≤ pre-allocated –Ī—É—Ą–Ķ—Ä (–Ī–Ķ–∑ malloc)
        auto fmt = [&buf, &pos](const char* key, double val) -> const char* {
            int n = snprintf(buf + pos, sizeof(buf) - pos, "%s,%.8f,", key, val);
            pos += n;
            return buf + pos - n;
        };

        // –í–ĺ–∑–≤—Ä–į—Č–į–Ķ–ľ —É–ļ–į–∑–į—ā–Ķ–Ľ–ł –Ĺ–į —Ā—ā—Ä–ĺ–ļ–ł –≤ –Ī—É—Ą–Ķ—Ä–Ķ
        // (–Ē–Ľ—Ź production –Ľ—É—á—ą–Ķ –ł—Ā–Ņ–ĺ–Ľ—Ć–∑–ĺ–≤–į—ā—Ć std::format –ł–Ľ–ł snprintf —Ā —ā–ĺ—á–Ĺ—č–ľ –ļ–ĺ–Ĺ—ā—Ä–ĺ–Ľ–Ķ–ľ)
        // –ó–ī–Ķ—Ā—Ć —É–Ņ—Ä–ĺ—Č–Ķ–Ĺ–ĺ –ī–Ľ—Ź –ī–Ķ–ľ–ĺ–Ĺ—Ā—ā—Ä–į—Ü–ł–ł –į—Ä—Ö–ł—ā–Ķ–ļ—ā—É—Ä—č
        std::string s_order(order_id), s_symbol(symbol), s_side(side);
        std::string s_price(std::to_string(price));
        std::string s_qty(std::to_string(qty));

        static thread_local const char* vals[5];
        static thread_local int lens[5];
        static thread_local int fmts[5] = {0,0,0,0,0}; // 0 = text format

        vals[0] = s_order.c_str(); lens[0] = s_order.size();
        vals[1] = s_symbol.c_str(); lens[1] = s_symbol.size();
        vals[2] = s_side.c_str(); lens[2] = s_side.size();
        vals[3] = s_price.c_str(); lens[3] = s_price.size();
        vals[4] = s_qty.c_str(); lens[4] = s_qty.size();

        DbParamBatch batch;
        batch.count = 5;
        batch.values = vals;
        batch.lengths = lens;
        batch.formats = fmts;

            batch_queue_.push({batch});
            cv_.notify_one();
        }
    }
};

    io_uring ring_;

    std::thread worker_;
    std::queue<DbQueryTask> queue_;
    std::mutex mtx_;
    std::atomic<bool> running_{true};
    int socket_fd_ = -1;
    bool conn_ok_ = false;
    bool reconnecting_ = false;

    // Non-blocking connection establishment
    bool init_connection(const std::string& conn_str) {
        conn_ = PQconnectStart(conn_str.c_str());
        if (!conn_) return false;

        int sock = PQsocket(conn_);
        if (sock < 0) return false;
        socket_fd_ = sock;
        PQsetnonblocking(conn_, 1);

        ConnStatusType status = PQconnectPoll(conn_);
        while (status == PGRES_POLLING_READING || status == PGRES_POLLING_WRITING) {
            struct pollfd pfd;
            pfd.fd = socket_fd_;
            pfd.events = (status == PGRES_POLLING_READING) ? POLLIN : POLLOUT;
            if (poll(&pfd, 1, 1000) <= 0) return false;
            status = PQconnectPoll(conn_);
        }
        conn_ok_ = (status == PGRES_CONNECTION_OK);
        return conn_ok_;
    }

    // Auto-reconnect with exponential backoff
    bool try_reconnect() {
        if (reconnecting_) return false;
        reconnecting_ = true;
        std::this_thread::sleep_for(std::chrono::milliseconds(50));
        reconnecting_ = false;
        return init_connection("host=postgres port=5432 user=trader password=secret dbname=hft_db connect_timeout=2 keepalives=1 keepalives_idle=5 keepalives_interval=1 keepalives_count=3");
    }

    bool submit_async(PGconn* conn, std::string_view sql, int nParams,
                      const char* const* values, const int* lengths, const int* formats,
                      std::function<void(PGresult*)> cb) {
        if (!conn || !conn_ok_) {
            if (try_reconnect()) { conn_ = conn; conn_ok_ = true; }
            else return false;
        }
        PGresult* res = PQsendQueryParams(conn_, sql.data(), nParams, nullptr,
                                          values, lengths, formats, 0);
        if (!res) return false;
        PQclear(res);
        return true;
    }

    // Process libpq async results
    void process_pg_results() {
        if (!conn_ || !conn_ok_) return;
        int sock = PQsocket(conn_);
        if (sock < 0) return;

        // Check socket readiness via io_uring completions would be here.
        // Simplified: poll for readiness before consuming
        struct pollfd pfd = {sock, POLLIN | POLLOUT, 0};
        if (poll(&pfd, 1, 0) > 0) {
            if (pfd.revents & POLLIN) PQconsumeInput(conn_);
            if (pfd.revents & POLLOUT) { /* libpq handles write buffer */ }
        }

        PGresult* res;
        while ((res = PQgetResult(conn_)) != nullptr) {
            PGresStatus status = PQresultStatus(res);
            if (status == PGRES_COMMAND_OK || status == PGRES_TUPLES_OK) {
                // Success: execute callback
                // In production: map to specific handlers
            } else if (status == PGRES_FATAL_ERROR) {
                fprintf(stderr, "[DB] Fatal: %s\n", PQresultErrorMessage(res));
                PQfinish(conn_);
                conn_ok_ = false;
            }
            PQclear(res);
        }
    }

    void worker_loop() {
        // io_uring setup
        io_uring_queue_init(128, &ring_, 0);

        while (running_.load(std::memory_order_relaxed)) {
            // 1. Submit queue tasks to libpq
            std::lock_guard<std::mutex> lock(mtx_);
            while (!queue_.empty()) {
                auto task = std::move(queue_.front());
                queue_.pop();

                // Format double->binary for HFT
                int n = task.nParams;
                int binary_formats[8] = {1,1,1,1,1,1,1,1};

                submit_async(task.conn, std::string_view(task.sql_buf, strlen(task.sql_buf)),
                             n, task.paramValues, task.paramLengths, binary_formats,
                             std::move(task.on_result));
            }

            // 2. Process async results
            process_pg_results();

            // 3. io_uring poll & wait (simplified for DB socket)
            struct io_uring_sqe* sqe = io_uring_get_sqe(&ring_);
            io_uring_prep_poll_add(sqe, socket_fd_, POLLIN | POLLOUT);
            io_uring_sqe_set_data(sqe, nullptr);
            io_uring_submit(&ring_);

            struct io_uring_cqe* cqe;
            unsigned head;
            io_uring_peek_cqe(&ring_, &cqe, &head);
            if (cqe) {
                io_uring_cqe_seen(&ring_, cqe);
                process_pg_results();
            } else {
                std::this_thread::yield();
            }
        }

        io_uring_queue_exit(&ring_);
        PQfinish(conn_);
    }

    IoringDbBackend() {
        io_uring_queue_init_params(128, &ring_, nullptr, IORING_SETUP_SQPOLL);
    }
    ~IoringDbBackend() { stop(); }

    void start(const std::string& conn_str) {
        init_connection(conn_str);
        worker_ = std::thread(&IoringDbBackend::worker_loop, this);
    }
    void stop() {
        running_.store(false, std::memory_order_release);
        if (worker_.joinable()) worker_.join();
    }

    void submit(PGconn* conn, std::string sql, int nParams,

        std::copy(formats, formats + nParams, task.paramFormats);
        task.on_result = std::move(cb);

        {
            std::lock_guard<std::mutex> lock(mtx_);
            queue_.push(std::move(task));
        }
    }
