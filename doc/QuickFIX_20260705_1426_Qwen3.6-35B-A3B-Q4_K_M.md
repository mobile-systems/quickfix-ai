#include "db_manager.hpp"
#include <stdexcept>

DbManager::DbManager(const std::string& conn_str) : m_conn_str(conn_str) {}

void DbManager::insertQuotes(const std::vector<Quote>& quotes) {
    pqxx::connection conn(m_conn_str);
    pqxx::work txn(conn);
    std::string sql = R"(
        INSERT INTO quotes (symbol, bid, ask, last, timestamp)
        VALUES ($1, $2, $3, $4, to_timestamp($5::bigint/1000000000))
    )";
    for (const auto& q : quotes) {
        txn.exec_prepared("q_ins", q.symbol, q.bid, q.ask, q.last, q.timestamp_ns);
    }
    txn.commit();
}

void DbManager::insertTrades(const std::vector<Trade>& trades) {
    pqxx::connection conn(m_conn_str);
    pqxx::work txn(conn);
    std::string sql = R"(
        INSERT INTO trades (order_id, symbol, side, ord_type, price, qty, exec_type, cum_qty, leaves_qty, timestamp)
        VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9, to_timestamp($10::bigint/1000000000))
    )";
    for (const auto& t : trades) {
        txn.exec_prepared("t_ins", t.order_id, t.symbol, t.side, t.ord_type, t.price, t.qty, t.exec_type, t.cum_qty, t.leaves_qty, t.timestamp_ns);
    }
    txn.commit();
}
