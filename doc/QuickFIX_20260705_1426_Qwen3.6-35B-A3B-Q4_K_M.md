### 1. `src/ring_buffer.hpp`
```cpp
#pragma once
#include <array>
#include <atomic>
#include <cstddef>
#include <cstdint>
#include <type_traits>

template<typename T, std::size_t Capacity>
requires (Capacity > 0) && ((Capacity & (Capacity - 1)) == 0) // power of 2
struct alignas(64) RingBuffer {
    static constexpr std::size_t Mask = Capacity - 1;
    static constexpr std::size_t CachePad = 64 / sizeof(T);

    alignas(64) std::atomic<std::size_t> tail{0};
    alignas(64) std::atomic<std::size_t> head{0};
    alignas(64) T buffer[Capacity];
    char pad[CachePad * sizeof(T) - sizeof(std::atomic<std::size_t>) * 2 - Capacity * sizeof(T)];

    [[nodiscard]] bool push(const T& val) {
        std::size_t t = tail.load(std::memory_order_relaxed);
        std::size_t next = (t + 1) & Mask;
        if (next == head.load(std::memory_order_acquire)) return false;
        buffer[t] = val;
        tail.store(next, std::memory_order_release);
        return true;
    }

    [[nodiscard]] bool pop(T& val) {
        std::size_t h = head.load(std::memory_order_relaxed);
        if (h == tail.load(std::memory_order_acquire)) return false;
        val = buffer[h];
        head.store((h + 1) & Mask, std::memory_order_release);
        return true;
    }
};
```

```cpp
    std::string_view raw;
    uint8_t msg_type;
    int32_t seq_num;
    int32_t length;
    uint32_t checksum;
    char reserved[32]; // padding
};

// Простой парсер FIX 4.4 для горячего пути
class FixParser {
public:
    static constexpr uint8_t MD_SNAPSHOT = 0x56; // 'V'
    static constexpr uint8_t EXEC_REPORT = 0x38; // '8'

    [[nodiscard]] static FixMessage parse(const char* buf, std::size_t len) {
        FixMessage msg{};
        msg.raw = std::string_view(buf, len);
        msg.checksum = 0;

        // Парсим MsgLength (tag 9)
        const char* p = buf;
        while (p < buf + len - 2) {
            if (p[0] == '9' && p[1] == '=') {
                msg.length = 0;
                p += 2;
                while (p < buf + len && *p >= '0' && *p <= '9') {
                    msg.length = msg.length * 10 + (*p++ - '0');
                }
                break;
            }
            ++p;
        }

        // Парсим MsgType (tag 35)
        p = buf;
        while (p < buf + len) {
            if (p[0] == '3' && p[1] == '=') {
                msg.msg_type = static_cast<uint8_t>(p[2]);
                break;
            }
            ++p;
        }

        // Парсим MsgSeqNum (tag 34)
        p = buf;
        while (p < buf + len) {
            if (p[0] == '3' && p[1] == '4' && p[2] == '=') {
                msg.seq_num = 0;
                p += 3;
                while (p < buf + len && *p >= '0' && *p <= '9') {
                    msg.seq_num = msg.seq_num * 10 + (*p++ - '0');
                }
                break;
            }
            ++p;
        }

        // CheckSum (tag 10)
        p = buf;
        while (p < buf + len - 7) { // 7 = len("10=000")
            if (p[0] == '1' && p[1] == '0' && p[2] == '=') {
                msg.checksum = 0;
                p += 3;
                while (p < buf + len && *p >= '0' && *p <= '9') {
                    msg.checksum = msg.checksum * 10 + (*p++ - '0');
                }
                break;
            }
            ++p;
        }
        return msg;
    }
};
```

```cpp
#pragma once
#include <sys/epoll.h>
#include <sys/socket.h>
#include <netinet/tcp.h>
#include <unistd.h>
#include <cstdint>
#include <cstring>
#include <string_view>
#include "ring_buffer.hpp"
#include "fix_message.hpp"

class TcpListener {
    int fd_ = -1;
    int epoll_ = -1;
    RingBuffer<FixMessage, 1 << 18> queue_;
    char buf_[1 << 16];
    std::size_t buf_pos_ = 0;
    int32_t pending_len_ = 0;

public:
    TcpListener(const char* addr, uint16_t port) {
        fd_ = socket(AF_INET, SOCK_STREAM, 0);
        int opt = 1;
        setsockopt(fd_, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
        setsockopt(fd_, IPPROTO_TCP, TCP_NODELAY, &opt, sizeof(opt));
        struct sockaddr_in sa{};
        sa.sin_family = AF_INET;
        sa.sin_port = htons(port);
        inet_pton(AF_INET, addr, &sa.sin_addr);
        bind(fd_, (struct sockaddr*)&sa, sizeof(sa));
        listen(fd_, 128);

        epoll_ = epoll_create1(EPOLL_CLOEXEC);
        struct epoll_event ev{};
        ev.events = EPOLLIN | EPOLLET;
        ev.data.fd = fd_;
        epoll_ctl(epoll_, EPOLL_CTL_ADD, fd_, &ev);
    }

    ~TcpListener() {
        if (fd_ >= 0) close(fd_);
        if (epoll_ >= 0) close(epoll_);
    }

    int run(std::size_t max_events = 4) {
        struct epoll_event events[16];
        while (true) {
            int n = epoll_wait(epoll_, events, 16, -1);
            for (int i = 0; i < n; ++i) {
                if (events[i].data.fd == fd_) {
                    struct sockaddr_in client;
                    socklen_t len = sizeof(client);
                    int cfd = accept(fd_, (struct sockaddr*)&client, &len);
                    int opt = 1;
                    setsockopt(cfd, IPPROTO_TCP, TCP_NODELAY, &opt, sizeof(opt));
                    struct epoll_event ev{};
                    ev.events = EPOLLIN | EPOLLET;
                    ev.data.fd = cfd;
                    epoll_ctl(epoll_, EPOLL_CTL_ADD, cfd, &ev);
                } else {
                    ssize_t n = recv(events[i].data.fd, buf_ + buf_pos_, sizeof(buf_) - buf_pos_, MSG_WAITALL);
                    if (n <= 0) { close(events[i].data.fd); continue; }
                    buf_pos_ += n;

                    // Простой state machine: ждем MsgLength, потом точное число байт
                    if (pending_len_ <= 0 && buf_pos_ >= 50) {
                        auto msg = FixParser::parse(buf_, buf_pos_);
                        if (msg.length > 0 && msg.length <= 65535) {
                            pending_len_ = msg.length;
                        }
                    }

                    if (pending_len_ > 0 && buf_pos_ >= pending_len_) {
                        auto msg = FixParser::parse(buf_, pending_len_);
                        queue_.push(msg);
                        // Сдвигаем буфер
                        memmove(buf_, buf_ + pending_len_, buf_pos_ - pending_len_);
                        buf_pos_ -= pending_len_;
                        pending_len_ = 0;
                    }
                }
            }
        }
        return 0;
    }
};
```

## 🛠 `src/db_async_param.hpp` (Параметрическая модель)

```cpp
#pragma once
#include <libpq-fe.h>
#include <thread>
#include <atomic>
#include <vector>
#include <queue>
#include <mutex>
#include <condition_variable>
#include <string_view>
#include <cstdint>
#include <cstring>
#include <chrono>
#include <iostream>

// Структура для передачи параметров без аллокаций
struct DbParamBatch {
    int count;
    const char* values[8]; // max 8 параметров (trade/quote/balance)
    int lengths[8];
    int formats[8];
};

class DbAsyncWriter {
    PGconn* conn_ = nullptr;
    std::thread worker_;
    std::atomic<bool> running_{true};

    std::queue<std::vector<DbParamBatch>> batch_queue_;
    std::mutex mtx_;
    std::condition_variable cv_;

    void init_connection() {
        std::string conn_str = "host=postgres port=5432 user=trader password=secret dbname=hft_db "
                               "connect_timeout=5 keepalives=1 keepalives_idle=10 keepalives_interval=5 keepalives_count=5";
        conn_ = PQconnectdb(conn_str.c_str());
        if (PQstatus(conn_) != CONNECTION_OK) {
            std::cerr << "[DB] Connect failed: " << PQerrorMessage(conn_) << "\n";
            std::exit(1);
        }
        // Неblocking режим для асинхронной обработки
        PQsetnonblocking(conn_, 1);

        // Подготовка statement один раз на подключение
        const char* stmt_trade = "PREPARE ins_trade (varchar, varchar, varchar, numeric, numeric) AS "
                                 "INSERT INTO trades (order_id, symbol, side, price, qty) VALUES ($1, $2, $3, $4, $5)";
        const char* stmt_quote = "PREPARE ins_quote (varchar, numeric, numeric, numeric) AS "
                                 "INSERT INTO quotes (symbol, bid, ask, last) VALUES ($1, $2, $3, $4)";
        const char* stmt_bal   = "PREPARE ins_bal (varchar, numeric, numeric) AS "
                                 "INSERT INTO balances (account, cash, equity) VALUES ($1, $2, $3)";

        PQexec(conn_, stmt_trade);
        PQexec(conn_, stmt_quote);
        PQexec(conn_, stmt_bal);
    }

    void flush_batch() {
        std::lock_guard<std::mutex> lock(mtx_);
        if (batch_queue_.empty()) return;

        auto batch = std::move(batch_queue_.front());
        batch_queue_.pop();

        // Выполняем параметры
        for (const auto& p : batch) {
            PGresult* res = nullptr;

            // Выбираем statement по кол-ву параметров
            if (p.count == 5) res = PQexecParams(conn_, "ins_trade", p.count, nullptr,
                                                  p.values, p.lengths, p.formats, 0);
            else if (p.count == 4) res = PQexecParams(conn_, "ins_quote", p.count, nullptr,
                                                       p.values, p.lengths, p.formats, 0);
            else if (p.count == 3) res = PQexecParams(conn_, "ins_bal", p.count, nullptr,
                                                       p.values, p.lengths, p.formats, 0);
            else continue;

            if (PQresultStatus(res) != PGRES_COMMAND_OK) {
                std::cerr << "[DB] Error: " << PQresultErrorMessage(res) << "\n";
            }
            PQclear(res);
        }
    }

    void worker_func() {
        init_connection();

        while (running_.load(std::memory_order_relaxed)) {
            std::unique_lock<std::mutex> lock(mtx_);
            cv_.wait_for(lock, std::chrono::milliseconds(2),
                         [this]{ return !batch_queue_.empty() || !running_.load(); });

            flush_batch();
        }

        PQfinish(conn_);
    }

public:
    DbAsyncWriter() {
        worker_ = std::thread(&DbAsyncWriter::worker_func, this);
    }
    ~DbAsyncWriter() {
        running_.store(false, std::memory_order_release);
        cv_.notify_one();
        worker_.join();
    }

    // Безопасный push с фиксированным пулом памяти
    void push_trade(const char* order_id, const char* symbol, const char* side,
                    double price, double qty) {
        static thread_local char buf[256];
        static thread_local int pos = 0;

        // Форматирование в pre-allocated буфер (без malloc)
        auto fmt = [&buf, &pos](const char* key, double val) -> const char* {
            int n = snprintf(buf + pos, sizeof(buf) - pos, "%s,%.8f,", key, val);
            pos += n;
            return buf + pos - n;
        };

        // Возвращаем указатели на строки в буфере
        // (Для production лучше использовать std::format или snprintf с точным контролем)
        // Здесь упрощено для демонстрации архитектуры
        std::string s_order(order_id), s_symbol(symbol), s_side(side);
        std::string s_price(std::to_string(price));
        std::string s_qty(std::to_string(qty));

        static thread_local const char* vals[5];
        static thread_local int lens[5];
        static thread_local int fmts[5] = {0,0,0,0,0}; // 0 = text format

        vals[0] = s_order.c_str(); lens[0] = s_order.size();
        vals[1] = s_symbol.c_str(); lens[1] = s_symbol.size();
        vals[2] = s_side.c_str(); lens[2] = s_side.size();
        vals[3] = s_price.c_str(); lens[3] = s_price.size();
        vals[4] = s_qty.c_str(); lens[4] = s_qty.size();

        DbParamBatch batch;
        batch.count = 5;
        batch.values = vals;
        batch.lengths = lens;
        batch.formats = fmts;

        {
            std::lock_guard<std::mutex> lock(mtx_);
            batch_queue_.push({batch});
            cv_.notify_one();
        }
    }
};
```

## ⚡ Производительность: параметрическая vs строковая

| Метрика | Строковая конкатенация | `PQexecParams` + Prepared |
|---------|------------------------|---------------------------|
| **План запроса** | Повторный parse каждый раз | Кэшируется сервером (`shared_buffers`/`pg_plan_cache`) |
| **Безопасность** | SQLi-риск, manual escaping | Полная защита, авто-конвертация типов |
| **CPU на hot-path** | `sprintf`/`std::string` аллокации | Прямой memcpy параметров |
| **Сеть** | Один большой текст | Заголовки + параметры (оптимизировано libpq) |
| **Результат** | ❌ Медленнее при частых вызовах | ✅ Стабильно быстрее, предсказуемо по latency |

> **Важно для HFT:** PostgreSQL оптимизирует prepared statements на уровне ядра. При частых вызовах с одинаковой структурой план кэшируется, и оверхед сводится к `memcpy` параметров. Строковая конкатенация даёт промахи кеша, дополнительную работу `std::string` и непредсказуемую latency.

## 🔧 Production-оптимизации (если уйдёте в strict HFT)

1. **Binary format (`format = 1`)**: `PQexecParams(..., 1)` передаёт double/int в бинарном виде. Экономит 20-40% сетевой нагрузки и CPU парсинга.
2. **`COPY` вместо `INSERT`**: Для >10k записей/сек используйте `PQcopyTuple` или временный файл/Shared Memory. `COPY` в 3-10x быстрее.
3. **`liburing` + `PQsocket`**: true async без thread-per-writer. `epoll` на `PQsocket()` + `PQgetResult()` loop.
4. **Connection pooling**: `PQconnectdbParams` + `libpq` async pool или `pgbouncer` в transaction-mode.
5. **NUMA & cache-line**: Выделите память для `DbParamBatch` через `mmap`/`posix_memalign` на NUMA-ноде с CPU worker.
