### 1. `src/ring_buffer.hpp`
```cpp
#pragma once
#include <array>
#include <atomic>
#include <cstddef>
#include <cstdint>
#include <type_traits>

template<typename T, std::size_t Capacity>
requires (Capacity > 0) && ((Capacity & (Capacity - 1)) == 0) // power of 2
struct alignas(64) RingBuffer {
    static constexpr std::size_t Mask = Capacity - 1;
    static constexpr std::size_t CachePad = 64 / sizeof(T);

    alignas(64) std::atomic<std::size_t> tail{0};
    alignas(64) std::atomic<std::size_t> head{0};
    alignas(64) T buffer[Capacity];
    char pad[CachePad * sizeof(T) - sizeof(std::atomic<std::size_t>) * 2 - Capacity * sizeof(T)];

    [[nodiscard]] bool push(const T& val) {
        std::size_t t = tail.load(std::memory_order_relaxed);
        std::size_t next = (t + 1) & Mask;
        if (next == head.load(std::memory_order_acquire)) return false;
        buffer[t] = val;
        tail.store(next, std::memory_order_release);
        return true;
    }

    [[nodiscard]] bool pop(T& val) {
        std::size_t h = head.load(std::memory_order_relaxed);
        if (h == tail.load(std::memory_order_acquire)) return false;
        val = buffer[h];
        head.store((h + 1) & Mask, std::memory_order_release);
        return true;
    }
};
```

```cpp
    std::string_view raw;
    uint8_t msg_type;
    int32_t seq_num;
    int32_t length;
    uint32_t checksum;
    char reserved[32]; // padding
};

// Простой парсер FIX 4.4 для горячего пути
class FixParser {
public:
    static constexpr uint8_t MD_SNAPSHOT = 0x56; // 'V'
    static constexpr uint8_t EXEC_REPORT = 0x38; // '8'

    [[nodiscard]] static FixMessage parse(const char* buf, std::size_t len) {
        FixMessage msg{};
        msg.raw = std::string_view(buf, len);
        msg.checksum = 0;

        // Парсим MsgLength (tag 9)
        const char* p = buf;
        while (p < buf + len - 2) {
            if (p[0] == '9' && p[1] == '=') {
                msg.length = 0;
                p += 2;
                while (p < buf + len && *p >= '0' && *p <= '9') {
                    msg.length = msg.length * 10 + (*p++ - '0');
                }
                break;
            }
            ++p;
        }

        // Парсим MsgType (tag 35)
        p = buf;
        while (p < buf + len) {
            if (p[0] == '3' && p[1] == '=') {
                msg.msg_type = static_cast<uint8_t>(p[2]);
                break;
            }
            ++p;
        }

        // Парсим MsgSeqNum (tag 34)
        p = buf;
        while (p < buf + len) {
            if (p[0] == '3' && p[1] == '4' && p[2] == '=') {
                msg.seq_num = 0;
                p += 3;
                while (p < buf + len && *p >= '0' && *p <= '9') {
                    msg.seq_num = msg.seq_num * 10 + (*p++ - '0');
                }
                break;
            }
            ++p;
        }

        // CheckSum (tag 10)
        p = buf;
        while (p < buf + len - 7) { // 7 = len("10=000")
            if (p[0] == '1' && p[1] == '0' && p[2] == '=') {
                msg.checksum = 0;
                p += 3;
                while (p < buf + len && *p >= '0' && *p <= '9') {
                    msg.checksum = msg.checksum * 10 + (*p++ - '0');
                }
                break;
            }
            ++p;
        }
        return msg;
    }
};
```

```cpp
#pragma once
#include <sys/epoll.h>
#include <sys/socket.h>
#include <netinet/tcp.h>
#include <unistd.h>
#include <cstdint>
#include <cstring>
#include <string_view>
#include "ring_buffer.hpp"
#include "fix_message.hpp"

class TcpListener {
    int fd_ = -1;
    int epoll_ = -1;
    RingBuffer<FixMessage, 1 << 18> queue_;
    char buf_[1 << 16];
    std::size_t buf_pos_ = 0;
    int32_t pending_len_ = 0;

public:
    TcpListener(const char* addr, uint16_t port) {
        fd_ = socket(AF_INET, SOCK_STREAM, 0);
        int opt = 1;
        setsockopt(fd_, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
        setsockopt(fd_, IPPROTO_TCP, TCP_NODELAY, &opt, sizeof(opt));
        struct sockaddr_in sa{};
        sa.sin_family = AF_INET;
        sa.sin_port = htons(port);
        inet_pton(AF_INET, addr, &sa.sin_addr);
        bind(fd_, (struct sockaddr*)&sa, sizeof(sa));
        listen(fd_, 128);

        epoll_ = epoll_create1(EPOLL_CLOEXEC);
        struct epoll_event ev{};
        ev.events = EPOLLIN | EPOLLET;
        ev.data.fd = fd_;
        epoll_ctl(epoll_, EPOLL_CTL_ADD, fd_, &ev);
    }

    ~TcpListener() {
        if (fd_ >= 0) close(fd_);
        if (epoll_ >= 0) close(epoll_);
    }

    int run(std::size_t max_events = 4) {
        struct epoll_event events[16];
        while (true) {
            int n = epoll_wait(epoll_, events, 16, -1);
            for (int i = 0; i < n; ++i) {
                if (events[i].data.fd == fd_) {
                    struct sockaddr_in client;
                    socklen_t len = sizeof(client);
                    int cfd = accept(fd_, (struct sockaddr*)&client, &len);
                    int opt = 1;
                    setsockopt(cfd, IPPROTO_TCP, TCP_NODELAY, &opt, sizeof(opt));
                    struct epoll_event ev{};
                    ev.events = EPOLLIN | EPOLLET;
                    ev.data.fd = cfd;
                    epoll_ctl(epoll_, EPOLL_CTL_ADD, cfd, &ev);
                } else {
                    ssize_t n = recv(events[i].data.fd, buf_ + buf_pos_, sizeof(buf_) - buf_pos_, MSG_WAITALL);
                    if (n <= 0) { close(events[i].data.fd); continue; }
                    buf_pos_ += n;

                    // Простой state machine: ждем MsgLength, потом точное число байт
                    if (pending_len_ <= 0 && buf_pos_ >= 50) {
                        auto msg = FixParser::parse(buf_, buf_pos_);
                        if (msg.length > 0 && msg.length <= 65535) {
                            pending_len_ = msg.length;
                        }
                    }

                    if (pending_len_ > 0 && buf_pos_ >= pending_len_) {
                        auto msg = FixParser::parse(buf_, pending_len_);
                        queue_.push(msg);
                        // Сдвигаем буфер
                        memmove(buf_, buf_ + pending_len_, buf_pos_ - pending_len_);
                        buf_pos_ -= pending_len_;
                        pending_len_ = 0;
                    }
                }
            }
        }
        return 0;
    }
};
```

## 🛠 `src/db_async_param.hpp` (Параметрическая модель)

```cpp
#pragma once
#include <libpq-fe.h>
#include <thread>
#include <atomic>
#include <vector>
#include <queue>
#include <mutex>
#include <condition_variable>
#include <string_view>
#include <cstdint>
#include <cstring>
#include <chrono>
#include <iostream>

// Структура для передачи параметров без аллокаций
struct DbParamBatch {
    int count;
    const char* values[8]; // max 8 параметров (trade/quote/balance)
    int lengths[8];
    int formats[8];
};

class DbAsyncWriter {
    PGconn* conn_ = nullptr;
    std::thread worker_;
    std::atomic<bool> running_{true};

    std::queue<std::vector<DbParamBatch>> batch_queue_;
    std::mutex mtx_;
    std::condition_variable cv_;

    void init_connection() {
        std::string conn_str = "host=postgres port=5432 user=trader password=secret dbname=hft_db "
                               "connect_timeout=5 keepalives=1 keepalives_idle=10 keepalives_interval=5 keepalives_count=5";
        conn_ = PQconnectdb(conn_str.c_str());
        if (PQstatus(conn_) != CONNECTION_OK) {
            std::cerr << "[DB] Connect failed: " << PQerrorMessage(conn_) << "\n";
            std::exit(1);
        }
        // Неblocking режим для асинхронной обработки
        PQsetnonblocking(conn_, 1);

        // Подготовка statement один раз на подключение
        const char* stmt_trade = "PREPARE ins_trade (varchar, varchar, varchar, numeric, numeric) AS "
                                 "INSERT INTO trades (order_id, symbol, side, price, qty) VALUES ($1, $2, $3, $4, $5)";
        const char* stmt_quote = "PREPARE ins_quote (varchar, numeric, numeric, numeric) AS "
                                 "INSERT INTO quotes (symbol, bid, ask, last) VALUES ($1, $2, $3, $4)";
        const char* stmt_bal   = "PREPARE ins_bal (varchar, numeric, numeric) AS "
                                 "INSERT INTO balances (account, cash, equity) VALUES ($1, $2, $3)";

        PQexec(conn_, stmt_trade);
        PQexec(conn_, stmt_quote);
        PQexec(conn_, stmt_bal);
    }

    void flush_batch() {
        std::lock_guard<std::mutex> lock(mtx_);
        if (batch_queue_.empty()) return;

        auto batch = std::move(batch_queue_.front());
        batch_queue_.pop();

        // Выполняем параметры
        for (const auto& p : batch) {
            PGresult* res = nullptr;

            // Выбираем statement по кол-ву параметров
            if (p.count == 5) res = PQexecParams(conn_, "ins_trade", p.count, nullptr,
                                                  p.values, p.lengths, p.formats, 0);
            else if (p.count == 4) res = PQexecParams(conn_, "ins_quote", p.count, nullptr,
                                                       p.values, p.lengths, p.formats, 0);
            else if (p.count == 3) res = PQexecParams(conn_, "ins_bal", p.count, nullptr,
                                                       p.values, p.lengths, p.formats, 0);
            else continue;

            if (PQresultStatus(res) != PGRES_COMMAND_OK) {
                std::cerr << "[DB] Error: " << PQresultErrorMessage(res) << "\n";
            }
            PQclear(res);
        }
    }

    void worker_func() {
        init_connection();

        while (running_.load(std::memory_order_relaxed)) {
            std::unique_lock<std::mutex> lock(mtx_);
            cv_.wait_for(lock, std::chrono::milliseconds(2),
                         [this]{ return !batch_queue_.empty() || !running_.load(); });

            flush_batch();
        }

        PQfinish(conn_);
    }

public:
    DbAsyncWriter() {
        worker_ = std::thread(&DbAsyncWriter::worker_func, this);
    }
    ~DbAsyncWriter() {
        running_.store(false, std::memory_order_release);
        cv_.notify_one();
        worker_.join();
    }

    // Безопасный push с фиксированным пулом памяти
    void push_trade(const char* order_id, const char* symbol, const char* side,
                    double price, double qty) {
        static thread_local char buf[256];
        static thread_local int pos = 0;

        // Форматирование в pre-allocated буфер (без malloc)
        auto fmt = [&buf, &pos](const char* key, double val) -> const char* {
            int n = snprintf(buf + pos, sizeof(buf) - pos, "%s,%.8f,", key, val);
            pos += n;
            return buf + pos - n;
        };

        // Возвращаем указатели на строки в буфере
        // (Для production лучше использовать std::format или snprintf с точным контролем)
        // Здесь упрощено для демонстрации архитектуры
        std::string s_order(order_id), s_symbol(symbol), s_side(side);
        std::string s_price(std::to_string(price));
        std::string s_qty(std::to_string(qty));

        static thread_local const char* vals[5];
        static thread_local int lens[5];
        static thread_local int fmts[5] = {0,0,0,0,0}; // 0 = text format

        vals[0] = s_order.c_str(); lens[0] = s_order.size();
        vals[1] = s_symbol.c_str(); lens[1] = s_symbol.size();
        vals[2] = s_side.c_str(); lens[2] = s_side.size();
        vals[3] = s_price.c_str(); lens[3] = s_price.size();
        vals[4] = s_qty.c_str(); lens[4] = s_qty.size();

        DbParamBatch batch;
        batch.count = 5;
        batch.values = vals;
        batch.lengths = lens;
        batch.formats = fmts;

        {
            std::lock_guard<std::mutex> lock(mtx_);
            batch_queue_.push({batch});
            cv_.notify_one();
        }
    }
};
```

## ⚡ Производительность: параметрическая vs строковая

| Метрика | Строковая конкатенация | `PQexecParams` + Prepared |
|---------|------------------------|---------------------------|
| **План запроса** | Повторный parse каждый раз | Кэшируется сервером (`shared_buffers`/`pg_plan_cache`) |
| **Безопасность** | SQLi-риск, manual escaping | Полная защита, авто-конвертация типов |
| **CPU на hot-path** | `sprintf`/`std::string` аллокации | Прямой memcpy параметров |
| **Сеть** | Один большой текст | Заголовки + параметры (оптимизировано libpq) |
| **Результат** | ❌ Медленнее при частых вызовах | ✅ Стабильно быстрее, предсказуемо по latency |

> **Важно для HFT:** PostgreSQL оптимизирует prepared statements на уровне ядра. При частых вызовах с одинаковой структурой план кэшируется, и оверхед сводится к `memcpy` параметров. Строковая конкатенация даёт промахи кеша, дополнительную работу `std::string` и непредсказуемую latency.

## 🔧 Production-оптимизации (если уйдёте в strict HFT)

1. **Binary format (`format = 1`)**: `PQexecParams(..., 1)` передаёт double/int в бинарном виде. Экономит 20-40% сетевой нагрузки и CPU парсинга.
2. **`COPY` вместо `INSERT`**: Для >10k записей/сек используйте `PQcopyTuple` или временный файл/Shared Memory. `COPY` в 3-10x быстрее.
3. **`liburing` + `PQsocket`**: true async без thread-per-writer. `epoll` на `PQsocket()` + `PQgetResult()` loop.
4. **Connection pooling**: `PQconnectdbParams` + `libpq` async pool или `pgbouncer` в transaction-mode.
5. **NUMA & cache-line**: Выделите память для `DbParamBatch` через `mmap`/`posix_memalign` на NUMA-ноде с CPU worker.

```cpp
    PGconn* conn;
    char sql_buf[512]; // Pre-allocated, no heap alloc on hot path
    int nParams;
    const char* paramValues[8];
    int paramLengths[8];
    int paramFormats[8]; // 0=text, 1=binary
    std::function<void(PGresult*)> on_result;

class IoringDbBackend {
    io_uring ring_;
    PGconn* conn_ = nullptr;
    std::thread worker_;
    std::queue<DbQueryTask> queue_;
    std::mutex mtx_;
    std::atomic<bool> running_{true};
    int socket_fd_ = -1;
    bool conn_ok_ = false;

    // Non-blocking connection establishment
    bool init_connection(const std::string& conn_str) {
        conn_ = PQconnectStart(conn_str.c_str());
        if (!conn_) return false;

        int sock = PQsocket(conn_);
        if (sock < 0) return false;
        socket_fd_ = sock;
        PQsetnonblocking(conn_, 1);

        ConnStatusType status = PQconnectPoll(conn_);
        while (status == PGRES_POLLING_READING || status == PGRES_POLLING_WRITING) {
            struct pollfd pfd;
            pfd.fd = socket_fd_;
            pfd.events = (status == PGRES_POLLING_READING) ? POLLIN : POLLOUT;
            if (poll(&pfd, 1, 1000) <= 0) return false;
            status = PQconnectPoll(conn_);
        }
        conn_ok_ = (status == PGRES_CONNECTION_OK);
        return conn_ok_;
    }

    // Auto-reconnect with exponential backoff
    bool try_reconnect() {
        if (reconnecting_) return false;
        reconnecting_ = true;
        std::this_thread::sleep_for(std::chrono::milliseconds(50));
        reconnecting_ = false;
        return init_connection("host=postgres port=5432 user=trader password=secret dbname=hft_db connect_timeout=2 keepalives=1 keepalives_idle=5 keepalives_interval=1 keepalives_count=3");
    }

    // Submit query to libpq async
    bool submit_async(PGconn* conn, std::string_view sql, int nParams,
                      const char* const* values, const int* lengths, const int* formats,
                      std::function<void(PGresult*)> cb) {
        if (!conn || !conn_ok_) {
            if (try_reconnect()) { conn_ = conn; conn_ok_ = true; }
            else return false;
        }
        PGresult* res = PQsendQueryParams(conn_, sql.data(), nParams, nullptr,
                                          values, lengths, formats, 0);
        if (!res) return false;
        PQclear(res);
        return true;
    }

    void process_pg_results() {
        if (!conn_ || !conn_ok_) return;
        int sock = PQsocket(conn_);
        if (sock < 0) return;

        // Check socket readiness via io_uring completions would be here.

        struct pollfd pfd = {sock, POLLIN | POLLOUT, 0};

            if (pfd.revents & POLLOUT) { /* libpq handles write buffer */ }
        }

        PGresult* res;
        while ((res = PQgetResult(conn_)) != nullptr) {
            PGresStatus status = PQresultStatus(res);
            if (status == PGRES_COMMAND_OK || status == PGRES_TUPLES_OK) {
                // Success: execute callback
                // In production: map to specific handlers
            } else if (status == PGRES_FATAL_ERROR) {
                fprintf(stderr, "[DB] Fatal: %s\n", PQresultErrorMessage(res));
                PQfinish(conn_);
                conn_ok_ = false;
            }
            PQclear(res);
        }
    }

    void worker_loop() {
        // io_uring setup
        io_uring_queue_init(128, &ring_, 0);

        while (running_.load(std::memory_order_relaxed)) {
            // 1. Submit queue tasks to libpq
            std::lock_guard<std::mutex> lock(mtx_);

                queue_.pop();

                // Format double->binary for HFT
                int n = task.nParams;

                submit_async(task.conn, std::string_view(task.sql_buf, strlen(task.sql_buf)),
                             n, task.paramValues, task.paramLengths, binary_formats,
                             std::move(task.on_result));
            }

            // 2. Process async results
            process_pg_results();

            // 3. io_uring poll & wait (simplified for DB socket)
            struct io_uring_sqe* sqe = io_uring_get_sqe(&ring_);
            io_uring_prep_poll_add(sqe, socket_fd_, POLLIN | POLLOUT);
            io_uring_sqe_set_data(sqe, nullptr);
            io_uring_submit(&ring_);

            struct io_uring_cqe* cqe;
            unsigned head;
            io_uring_peek_cqe(&ring_, &cqe, &head);
            if (cqe) {
                io_uring_cqe_seen(&ring_, cqe);
                process_pg_results();
            } else {
                std::this_thread::yield();
            }
        }

        io_uring_queue_exit(&ring_);
        PQfinish(conn_);
    }

public:
    IoringDbBackend() {
        io_uring_queue_init_params(128, &ring_, nullptr, IORING_SETUP_SQPOLL);
    }
    ~IoringDbBackend() { stop(); }

    void start(const std::string& conn_str) {
        init_connection(conn_str);
        worker_ = std::thread(&IoringDbBackend::worker_loop, this);
    }
    void stop() {
        running_.store(false, std::memory_order_release);
        if (worker_.joinable()) worker_.join();
    }

                const char* const* values, const int* lengths, const int* formats,
                std::function<void(PGresult*)> cb) {
        DbQueryTask task{};
        task.conn = conn;
        strncpy(task.sql_buf, sql.c_str(), sizeof(task.sql_buf) - 1);
        task.nParams = nParams;

        std::copy(lengths, lengths + nParams, task.paramLengths);
        std::copy(formats, formats + nParams, task.paramFormats);
        task.on_result = std::move(cb);

        {
            std::lock_guard<std::mutex> lock(mtx_);
            queue_.push(std::move(task));
        }
    }
```

## 📦 `src/uring_fix_listener.hpp`
```cpp
#pragma once
#include <liburing.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <netinet/tcp.h>
#include <unistd.h>
#include <cstring>
#include <cstdint>
#include <atomic>
#include <thread>
#include <array>
#include "ring_buffer.hpp"
#include "fix_message.hpp"

struct ConnCtx {
    int fd = -1;
    char buf[1 << 16];
    size_t pos = 0;
    int32_t pending_len = 0;
    uint8_t state = 0; // 0: WAIT_LEN, 1: READING_MSG, 2: PARSING
};

class UringFixListener {
    io_uring ring_;
    int server_fd_ = -1;
    std::array<ConnCtx*, 8> conns_;
    std::atomic<bool> running_{false};
    RingBuffer<FixMessage, 1 << 18> out_queue_;
    std::thread worker_;
    uint64_t sqe_seq_ = 0;

    void submit_accept();
    void submit_read(ConnCtx* ctx);
    void process_accept(struct io_uring_cqe* cqe);
    void process_read(struct io_uring_cqe* cqe, ConnCtx* ctx);
    void worker_loop();

public:
    UringFixListener(uint16_t port) {
        // IORING_SETUP_SQPOLL: kernel thread polls SQ, снижает latency до ~0.2μs
        io_uring_queue_init(256, &ring_, IORING_SETUP_SQPOLL | IORING_SETUP_IOPOLL);
        for (auto& c : conns_) c = new ConnCtx();

        server_fd_ = socket(AF_INET, SOCK_STREAM, 0);
        int opt = 1;
        setsockopt(server_fd_, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
        struct sockaddr_in sa{};
        sa.sin_family = AF_INET;
        sa.sin_port = htons(port);
        inet_pton(AF_INET, "0.0.0.0", &sa.sin_addr);
        bind(server_fd_, (struct sockaddr*)&sa, sizeof(sa));
        listen(server_fd_, 128);
    }

    ~UringFixListener() {
        stop();
        if (server_fd_ >= 0) shutdown(server_fd_, SHUT_RDWR);
        if (server_fd_ >= 0) close(server_fd_);
        for (auto c : conns_) delete c;
        io_uring_queue_exit(&ring_);
    }

    void start() {
        running_.store(true, std::memory_order_release);
        worker_ = std::thread(&UringFixListener::worker_loop, this);
        submit_accept();
    }

    void stop() {
        running_.store(false, std::memory_order_release);
        if (server_fd_ >= 0) shutdown(server_fd_, SHUT_RDWR);
        if (worker_.joinable()) worker_.join();
    }

    RingBuffer<FixMessage, 1 << 18>& get_out_queue() { return out_queue_; }

private:
    void submit_accept() {
        struct io_uring_sqe* sqe = io_uring_get_sqe(&ring_);
        if (!sqe) { io_uring_submit(&ring_); sqe = io_uring_get_sqe(&ring_); }
        struct sockaddr_in addr{};
        socklen_t addrlen = sizeof(addr);
        io_uring_prep_accept(sqe, server_fd_, (struct sockaddr*)&addr, &addrlen, 0);
        sqe->user_data = 0; // 0 = accept event
        io_uring_submit(&ring_);
    }

    void submit_read(ConnCtx* ctx) {
        struct io_uring_sqe* sqe = io_uring_get_sqe(&ring_);
        if (!sqe) { io_uring_submit(&ring_); sqe = io_uring_get_sqe(&ring_); }
        io_uring_prep_read(sqe, ctx->fd, ctx->buf, sizeof(ctx->buf), 0);
        // Для сокетов sqe->addr переопределяет адрес буфера, off игнорируется
        sqe->addr = (uint64_t)(ctx->buf + ctx->pos);
        sqe->len = sizeof(ctx->buf) - ctx->pos;
        sqe->user_data = (uint64_t)ctx | (1ULL << 48); // Tag for read events
        sqe->flags = IOSQE_FIXED_FILE;
        io_uring_submit(&ring_);
    }

    void process_accept(struct io_uring_cqe* cqe) {
        int new_fd = cqe->res;
        if (new_fd < 0) {
            if (new_fd == -EAGAIN) return;
            return;
        }
        ConnCtx* ctx = conns_[0];
        ctx->fd = new_fd;
        ctx->pos = 0;
        ctx->pending_len = 0;
        int opt = 1;
        setsockopt(new_fd, IPPROTO_TCP, TCP_NODELAY, &opt, sizeof(opt));
        submit_read(ctx);
        submit_accept();
    }

    void process_read(struct io_uring_cqe* cqe, ConnCtx* ctx) {
        ssize_t ret = cqe->res;
        if (ret < 0) {
            if (ret == -EAGAIN || ret == -EWOULDBLOCK) {
                // Нет данных, ждем следующее событие
                return;
            }
            close(ctx->fd);
            ctx->fd = -1;
            return;
        }
        ctx->pos += ret;

        // FIX 4.4 state machine (zero-copy, cache-line friendly)
        if (ctx->pending_len <= 0 && ctx->pos >= 50) {
            auto msg = FixParser::parse(ctx->buf, ctx->pos);
            if (msg.length > 0 && msg.length <= 65535) {
                ctx->pending_len = msg.length;
            }
        }

        if (ctx->pending_len > 0 && ctx->pos >= (size_t)ctx->pending_len) {
            auto msg = FixParser::parse(ctx->buf, ctx->pending_len);
            if (!out_queue_.push(msg)) {
                // Backpressure: в продакшене throttle or drop oldest
            }
            memmove(ctx->buf, ctx->buf + ctx->pending_len, ctx->pos - ctx->pending_len);
            ctx->pos -= ctx->pending_len;
            ctx->pending_len = 0;
        }

        if (ctx->pos < sizeof(ctx->buf)) {
            submit_read(ctx);
        }
    }

    void worker_loop() {
        while (running_.load(std::memory_order_acquire)) {
            struct io_uring_cqe* cqe;
            int ret = io_uring_wait_cqe(&ring_, &cqe);
            if (ret < 0 && ret != -EINTR) continue;

            if (cqe->res < 0 && cqe->res != -EAGAIN) {
                io_uring_cqe_seen(&ring_, cqe);
                continue;
            }

            uint64_t ud = cqe->user_data;
            if (ud == 0) process_accept(cqe);
            else process_read(cqe, (ConnCtx*)(ud & ((1ULL << 48) - 1)));

            io_uring_cqe_seen(&ring_, cqe);
        }
    }
};
```

## 📦 `src/main.cpp` (Интеграция)
```cpp
#include <iostream>
#include <sched.h>
#include <sys/resource.h>
#include "uring_fix_listener.hpp"
#include "db_async_uring.hpp"

static void pin_cpu(int core) {
    cpu_set_t cpuset;
    CPU_ZERO(&cpuset);
    CPU_SET(core, &cpuset);
    sched_setaffinity(0, sizeof(cpu_set_t), &cpuset);
}

static void set_mlockall() {
    if (mlockall(MCL_CURRENT | MCL_FUTURE) != 0) {
        std::cerr << "[WARN] mlockall failed: " << strerror(errno) << "\n";
    }
}

int main() {
    pin_cpu(0);
    set_mlockall();

    UringFixListener listener(9876);
    IoringDbBackend db_backend;
    db_backend.start("host=postgres port=5432 user=trader password=secret dbname=hft_db connect_timeout=2");

    listener.start();

    while (true) {
        FixMessage msg;
        if (listener.get_out_queue().pop(msg)) {
            // Dispatch по типам FIX 4.4
            if (msg.msg_type == FixParser::EXEC_REPORT) {
                char sql[256];
                snprintf(sql, sizeof(sql),
                         "INSERT INTO trades (order_id, symbol, side, price, qty) VALUES ($1, $2, $3, $4, $5)");
                static thread_local char price_buf[32], qty_buf[32];
                snprintf(price_buf, sizeof(price_buf), "%.8f", 1.08543210);
                snprintf(qty_buf, sizeof(qty_buf), "%.0f", 100000.0);

                const char* vals[5] = {"ORD-12345", "EURUSD", "B", price_buf, qty_buf};
                int lens[5] = {9, 6, 1, 10, 7};
                int fmt[5] = {1, 1, 1, 1, 1}; // Binary HFT mode

                db_backend.submit(db_backend.get_conn(), std::string(sql), 5, vals, lens, fmt,
                                  [](PGresult* r) { if (PQresultStatus(r) == PGRES_COMMAND_OK) /* ACK */; });
            }
        } else {
            std::this_thread::yield();
        }
    }

    listener.stop();
    db_backend.stop();
    return 0;
}
```
