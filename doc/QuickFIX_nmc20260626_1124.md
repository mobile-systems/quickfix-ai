## 2. Минимальный пример кода

```cpp
#include <quickfix/FileLog.h>
#include <quickfix/Message.h>
#include <quickfix/MessageCracker.h>
#include <quickfix/Logger.h>
#include <quickfix/SessionManager.h>

#include <quickfix/Callback.h>
#include <quickfix/definitions.h>

#include <pqxx/pqxx>

#include <iostream>
#include <string>
#include <chrono>
#include <thread>

/* -------------------------------------------------
   Простая структура, которая извлекает EURUSD из сообщения FIX и
   записывает строку в таблицу PostgreSQL.
   ------------------------------------------------- */
class QuoteStorage : public Callback {
public:
    QuoteStorage(const std::string& connStr) : connStr_(connStr) {}

    /* Эта функция вызывается, когда QuickFIX получает сообщение */
    void onMessage(const std::string& msg) override
    {
        // QuickFIX уже парсит сообщение в класс Message.
        // В данном примере мы получаем сообщение из строки, а затем
        // извлекаем поля.
        try
        {
            // 1) Парсинг сообщения FIX
            QuickFix::Message fixMsg;
            QuickFix::DataDictionary dd;
            dd.loadFieldTypes("spec/0005.xml"); // необязательно, можно опустить

            // Простой парсер, предполагающий, что сообщение находится в формате
            // "8=FIX.4.4|9=1234|35=D|55=EURUSD|54=0|60=20250517-12:00:00|75=1.1800|77=1.1810"
            // (тег 35=D = MarketDataSnapshot)
            // В реальном приложении вы бы использовали QuickFIX::Message::set
            // с помощью QuickFix::StringTokenizer.
            std::istringstream stream(msg);
            std::string line;
            while (std::getline(stream, line, '|'))
            {
                size_t pos = line.find('=');
                if (pos != std::string::npos)
                {
                    int tag = std::stoi(line.substr(0, pos));
                    std::string value = line.substr(pos + 1);
                    fixMsg.setField(tag, value);
                }
            }

            // 2) Извлечение полей, которые нам нужны
            QuickFix::Symbol sym;
            QuickFix::Price price;
            QuickFix::DateTime timeStamp;

            // QuickFix использует стандартную схему FIX; тег 55 = Symbol, 75 = LastPrice, 60 = TransactTime
            if (fixMsg.getField(sym) && sym == "EURUSD")
            {
                fixMsg.getField(price);
                fixMsg.getField(timeStamp);
            }
            else
            {
                std::cerr << "Сообщение FIX не содержит EURUSD, пропускаем." << std::endl;
                return;
            }

            // 3) Соединение с PostgreSQL и вставка данных
            try
            {
                // Подключение (используйте pqxx::connection {connStr})
                pqxx::connection conn(connStr_);

                // Открытие транзакции (обеспечивает atomicity)
                pqxx::transaction tx(conn);

                // Подготовленный запрос для вставки
                pqxx::prepared_stmt prep = tx.prepare(
                    "insert_eurusd",
                    "INSERT INTO eur_usd_quotes (sym, price, timestamp) "
                    "VALUES ($1, $2, $3) ON CONFLICT (timestamp) DO NOTHING"
                );

                // Преобразование timestamp в строковое представление, подходящее для PostgreSQL
                std::string tsStr = timeStamp.toString();

                tx.execute(prep, { sym, price, tsStr });

                std::cout << "Записан котировочный момент: " << tsStr << " " << price << std::endl;
            }
            catch (const std::exception& e)
            {
                std::cerr << "Ошибка PostgreSQL: " << e.what() << std::endl;
            }
        }
        catch (const std::exception& e)
        {
            std::cerr << "Ошибка обработки FIX: " << e.what() << std::endl;
        }
    }

private:
    std::string connStr_;
};

/* -------------------------------------------------
   Вспомогательная функция для имитации получения сообщения FIX.
   В реальном приложении сообщение FIX было бы получено через
   ThreadedSocketInitiator / ConnectionHandler.
   ------------------------------------------------- */
void simulateIncomingMessage(QuoteStorage& storage, const std::string& msg)
{
    storage.onMessage(msg);
}
```
