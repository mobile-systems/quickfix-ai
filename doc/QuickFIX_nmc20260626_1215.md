### include/fix_helper.hpp

```cpp
#pragma once

#include <quickfix/FileLog.h>
#include <quickfix/Logger.h>
#include <quickfix/Messages.h>
#include <quickfix/Application.h>
#include <quickfix/Initiator.h>
#include <quickfix/Connector.h>

#include <quickfix/SessionManager.h>
#include <quickfix/SessionSettings.h>
#include <quickfix/Callbacks.h>
#include <quickfix/MessageStore.h>
#include <quickfix/FileMessageStore.h>
#include <quickfix/Signal.h>
#include <quickfix/Exception.h>

#include <pqxx/pqxx>
#include <unordered_map>
#include <chrono>
#include <iomanip>
#include <sstream>
#include <iostream>
#include <thread>
#include <atomic>
#include <mutex>

// -----------------------------------------------------------------
// 1) Класс, который обрабатывает входящее сообщение FIX
// -----------------------------------------------------------------
class MarketDataApp : public QUICKFIX::Application {
public:
    // Вызывается, когда мы получаем Snapshot или Increment.
    // Мы обрабатываем только MDEntries с securityType = "CASH" и symbol = "EUR/USD"
    virtual void onMessage(const QUICKFIX::MarketDataSnapshot& msg) {
        process(msg);
    }

    virtual void onMessage(const QUICKFIX::MarketDataIncremental) {
        // Для упрощения мы обрабатываем только snapshot, но в продакшене вам, вероятно,
        // захотите использовать Incremental для экономии трафика.
        // Вы можете добавить аналогичную обработку здесь.
    }

    // Этот метод вызывается, когда сессия была успешно установлена (необязательно)
    virtual void onLogon(const QUICKFIX::Session& session) {
        std::cout << "[LOG] Сессия " << session.getSessionID() << " установлена." << std::endl;
    }

    virtual void onLogout(const QUICKFIX::Session& session) {
        std::cout << "[LOG] Сессия " << session.getSessionID() << " завершена." << std::endl;
    }

    // Получить последний обработанную котировку
    std::optional<MarketTick> lastTick() const {
        std::lock_guard<std::mutex> lock(mutex_);
        return last_tick_;
    }

private:
    void process(const QUICKFIX::MarketDataSnapshot& msg) {
        // 1) Извлечь price-level entries
        //    msg.getSymbol() -> std::string
        //    msg.getMDEntries() -> std::vector<QUICKFIX::MDEntry>
        //    Для каждого MDEntry проверить MDEntryType (Bid = '1', Ask = '2')
        // 2) Собрать структуру tick
        // 3) Сохранить в БД (асинхронно, через поток)
        // 4) Обновить last_tick_

        auto symbol = msg.getSymbol();
        // Ожидаем, что символ уже приведен к виду "EUR/USD"
        if (symbol != "EUR/USD") return;

        std::vector<QUICKFIX::MDEntry> bids, asks;
        for (const auto& e : msg.getMDEntries()) {
            if (e.getMDEntryType() == QUICKFIX::MDEntryType_Bid)
                bids.push_back(e);
            else if (e.getMDEntryType() == QUICKFIX::MDEntryType_Ask)
                asks.push_back(e);
        }

        if (bids.empty() || asks.empty()) {
            std::cerr << "[WARN] Пропущена котirovka без бид/аск." << std::endl;
            return;
        }

        // Берем первый уровень для простоты
        double bid = std::stod(bids[0].getPrice().c_str());
        double ask = std::stod(asks[0].getPrice().c_str());

        // Время FIX хранится в тикалке (ms since epoch)
        auto now = std::chrono::system_clock::now();
        auto epoch = now.epoch_realtime();
        // Конвертировать в PostgreSQL timestamp (ISO8601)
        auto tick = MarketTick{
            .symbol = symbol,
            .bid = bid,
            .ask = ask,
            .timestamp = std::to_string(epoch.count()) // упрощенно
        };

        {
            std::lock_guard<std::mutex> lock(mutex_);
            last_tick_ = std::move(tick);
        }

        // Сохранить в PostgreSQL в фоновом потоке (см. save_to_pg)
        std::thread t(&MarketDataApp::save_to_pg, this, tick);
        t.detach();
    }

    // -----------------------------------------------------------------
    // 2) Подключение к PostgreSQL и сохранение
    // -----------------------------------------------------------------
    void save_to_pg(const MarketTick& tick) {
        // Ожидаем, что пользователь предоставил URL соединения, либо через
        // переменную окружения либо через файл настроек.
        const char* conn_str = std::getenv("PG_CONN_STR");
        if (!conn_str) {
            std::cerr << "[ERROR] Ошибка: не указана строка соединения PostgreSQL (PG_CONN_STR)." << std::endl;
            return;
        }

        try {
            pqxx::connection conn(conn_str);
            pqxx::work tx(conn);

            // Простая вставка – избегайте SQL-инъекций с помощью prepared statements
            tx.exec_params(
                "INSERT INTO market_ticks (symbol, bid, ask, \"timestamp\") "
                "VALUES ($1,$2,$3,$4)",
                tick.symbol,
                tick.bid,
                tick.ask,
                tick.timestamp
            );

            tx.commit();
            std::cout << "[DB] Сохранена котirovka " << tick.symbol << " " << tick.bid << "/" << tick.ask << std::endl;
        } catch (const std::exception& e) {
            std::cerr << "[ERROR] Ошибка базы данных: " << e.what() << std::endl;
        }
    }

    // -----------------------------------------------------------------
    // 3) Утилиты
    // -----------------------------------------------------------------
    std::mutex mutex_;
    std::optional<MarketTick> last_tick_;
};

// -----------------------------------------------------------------
// Определения структур и типов
// -----------------------------------------------------------------
struct MarketTick {
    std::string symbol;
    double      bid;
    double      ask;
    std::string timestamp;
};

// -----------------------------------------------------------------
// Вспомогательные функции для инициализации сессии QuickFIX
// -----------------------------------------------------------------
static QUICKFIX::SessionSettings makeSettings(const std::string& config_file) {
    QUICKFIX::SessionSettings settings;
    // Обычный INI-файл, содержащий параметры сессии, таких как
    // ConnectionPoint = incoming | outgoing
    // SenderCompID / TargetCompID и т.д.
    // Для демонстрации мы просто используем пустые настройки.
    // В реальном приложении вы захотите загрузить конфигурацию из файла.
    return settings;
}
```

```cpp
// Класс приложения MarketDataApp ---------------------------------------------------------
class MarketDataApp : public QUICKFIX::Application {
public:
    // Обработчики сообщений
    void onMessage(const QUICKFIX::MarketDataSnapshot& msg) { process(msg); }
    void onMessage(const QUICKFIX::MarketDataIncremental& msg) { /* необязательно */ }

    void onLogon(const QUICKFIX::Session& session) {
        std::cout << "[LOG] Сессия " << session.getSessionID() << " установлена." << std::endl;
    }
    void onLogout(const QUICKFIX::Session& session) {
        std::cout << "[LOG] Сессия " << session.getSessionID() << " завершена." << std::endl;
    }

    // Получить последний обработанный тик
    std::optional<MarketTick> lastTick() const {
        std::lock_guard<std::mutex> lock(mutex_);
        return last_tick_;
    }

private:
    // Внутренняя логика ---------------------------------------------------------
    void process(const QUICKFIX::MarketDataSnapshot& msg) {
        auto symbol = msg.getSymbol();
        if (symbol != "EUR/USD") return; // Фильтрация по инструменту

        std::vector<QUICKFIX::MDEntry> bids, asks;
        for (const auto& e : msg.getMDEntries()) {
            if (e.getMDEntryType() == QUICKFIX::MDEntryType_Bid)
                bids.push_back(e);
            else if (e.getMDEntryType() == QUICKFIX::MDEntryType_Ask)
                asks.push_back(e);
        }

        if (bids.empty() || asks.empty()) {
            std::cerr << "[WARN] Пропущена котirovka без бид/аск.\n";
            return;
        }

        double bid  = std::stod(bids[0].getPrice().c_str());
        double ask  = std::stod(asks[0].getPrice().c_str());

        // Простое временное значение – вы можете использовать std::chrono::system_clock для лучшего результата
        auto now = std::chrono::system_clock::now();
        auto epoch = std::chrono::system_clock::to_time_t(now);
        std::ostringstream ts;
        ts << std::put_time(std::localtime(&epoch), "%Y-%m-%d %H:%M:%S");

        MarketTick tick{
            .symbol    = symbol,
            .bid        = bid,
            .ask        = ask,
            .timestamp = ts.str()
        };

        {
            std::lock_guard<std::mutex> lock(mutex_);
            last_tick_ = std::move(tick);
        }

        // Асинхронная запись в БД (используйте std::async или тредпул в реальном приложении)
        std::thread t(&MarketDataApp::save_to_pg, this, tick);
        t.detach();
    }

    void save_to_pg(const MarketTick& tick) {
        const char* conn_str = std::getenv("PG_CONN_STR");
        if (!conn_str) {
            std::cerr << "[ERROR] PG_CONN_STR не установлен.\n";
            return;
        }

        try {
            pqxx::connection conn(conn_str);
            pqxx::work tx(conn);
            tx.exec_params(
                "INSERT INTO market_ticks (symbol, bid, ask, \"timestamp\") "
                "VALUES ($1,$2,$3,$4)",
                tick.symbol,
                tick.bid,
                tick.ask,
                tick.timestamp
            );
            tx.commit();
            std::cout << "[DB] Сохранена котirovka " << tick.symbol
                      << " " << tick.bid << "/" << tick.ask << "\n";
        } catch (const std::exception& e) {
            std::cerr << "[ERROR] Ошибка базы данных: " << e.what() << "\n";
        }
    }

    std::mutex           mutex_;
    std::optional<MarketTick> last_tick_;
};
```

## 4. include/market_data_app.hpp
```cpp
#pragma once

#include "quickfix_eur_usd_all.hpp"

#include <unordered_map>
#include <mutex>
#include <thread>
#include <atomic>
#include <chrono>
#include <sstream>
#include <iomanip>
#include <iostream>
#include <stdexcept>
#include <vector>
#include <algorithm>

class MarketDataApp : public QUICKFIX::Application {
public:
    MarketDataApp(const QUICKFIX::SessionSettings& settings)
        : settings_(settings), running_(false) {}

    // Запустить получение котировок в отдельном потоке
    void start() {
        if (running_) return;
        running_ = true;
        worker_ = std::thread(&MarketDataApp::run, this);
    }

    // Остановиться
    void stop() {
        running_ = false;
        if (worker_.joinable()) worker_.join();
        if (initiator_) {
            initiator_->stop();
            initiator_.reset();
        }
    }

    // Получить последний тик для символа (если он обрабатывался)
    std::optional<MarketTick> getTick(const std::string& symbol) const {
        std::lock_guard<std::mutex> lock(ticks_mutex_);
        auto it = ticks_.find(symbol);
        return it != ticks_.end() ? std::optional<MarketTick>{it->second} : std::nullopt;
    }

    // Получить все текущие тики
    std::unordered_map<std::string, MarketTick> getAllTicks() const {
        std::lock_guard<std::mutex> lock(ticks_mutex_);
        return ticks_;
    }

private:
    // ----- QuickFIX Application -----
    void onMessage(const QUICKFIX::MarketDataSnapshot& msg) override {
        processSnapshot(msg);
    }

    void onMessage(const QUICKFIX::MarketDataIncremental& msg) override {
        // Для упрощения игнорируем инкрементальные сообщения.
    }

    void onLogon(const QUICKFIX::Session& session) override {
        std::cout << "[LOG] Сессия " << session.getSessionID() << " установлена.\n";
    }

    void onLogout(const QUICKFIX::Session& session) override {
        std::cout << "[LOG] Сессия " << session.getSessionID() << " завершена.\n";
        scheduleReconnect();
    }

    // ----- Вспомогательные функции -----
    void scheduleReconnect() {
        reconnect_timer_ = std::chrono::steady_clock::now() + reconnect_delay_;
    }

    void processSnapshot(const QUICKFIX::MarketDataSnapshot& msg) {
        auto symbol = msg.getSymbol();
        // Фильтровать только инструменты, которые мы отслеживаем
        if (std::find(instruments.begin(), instruments.end(), symbol) == instruments.end())
            return;

        std::vector<QUICKFIX::MDEntry> bids, asks;
        for (const auto& e : msg.getMDEntries()) {
            if (e.getMDEntryType() == QUICKFIX::MDEntryType_Bid)
                bids.push_back(e);
            else if (e.getMDEntryType() == QUICKFIX::MDEntryType_Ask)
                asks.push_back(e);
        }

        if (bids.empty() || asks.empty())
            return;

        double bid  = std::stod(bids[0].getPrice().c_str());
        double ask  = std::stod(asks[0].getPrice().c_str());

        auto now = std::chrono::system_clock::now();
        auto epoch = std::chrono::system_clock::to_time_t(now);
        std::ostringstream ts;
        ts << std::put_time(std::localtime(&epoch), "%Y-%m-%d %H:%M:%S");

        MarketTick tick{
            .symbol    = symbol,
            .bid        = bid,
            .ask        = ask,
            .timestamp = ts.str()
        };

        {
            std::lock_guard<std::mutex> lock(ticks_mutex_);
            ticks_[symbol] = std::move(tick);
        }

        // Асинхронная запись в базу данных
        std::thread t(&MarketDataApp::save_to_pg, this, tick);
        t.detach();
    }

    void save_to_pg(const MarketTick& tick) {
        const char* conn_str = std::getenv("PG_CONN_STR");
        if (!conn_str) {
            std::cerr << "[ERROR] PG_CONN_STR не установлен.\n";
            return;
        }

        try {
            pqxx::connection conn(conn_str);
            pqxx::work tx(conn);
            tx.exec_params(
                "INSERT INTO market_ticks (symbol, bid, ask, \"timestamp\") "
                "VALUES ($1,$2,$3,$4)",
                tick.symbol,
                tick.bid,
                tick.ask,
                tick.timestamp
            );
            tx.commit();
            std::cout << "[DB] Сохранена котirovka " << tick.symbol
                      << " " << tick.bid << "/" << tick.ask << "\n";
        } catch (const std::exception& e) {
            std::cerr << "[ERROR] Ошибка базы данных: " << e.what() << "\n";
        }
    }

    void run() {
        while (running_) {
            try {
                if (!initiator_) {
                    auto logFactory = std::make_shared<QUICKFIX::FileLogFactory>("log.txt");
                    auto msgStoreFactory = std::make_shared<QUICKFIX::FileMessageStoreFactory>("store.cfg");

                    initiator_ = std::make_unique<QUICKFIX::ThreadedSocketInitiator>(
                        std::static_pointer_cast<QUICKFIX::Application>(shared_from_this()), // мы должны иметь shared_ptr к этому объекту
                        settings_,
                        logFactory,
                        msgStoreFactory
                    );

                    // Запускаем Initiator в его собственном потоке.
                    initiator_->start();

                    std::cout << "[INIT] QuickFIX инициатор запущен.\n";
                }

                // Подождать до следующей итерации
                std::this_thread::sleep_for(std::chrono::seconds(1));

                // Проверить таймер переподключения
                auto now = std::chrono::steady_clock::now();
                if (reconnect_timer_ && now >= reconnect_timer_.value()) {
                    // Сбросить инициатор для нового переподключения
                    if (initiator_) {
                        initiator_->stop();
                        initiator_.reset();
                    }
                    reconnect_timer_.reset();
                }
            } catch (const std::exception& e) {
                std::cerr << "[QF] Ошибка: " << e.what() << "\n";
                std::this_thread::sleep_for(std::chrono::seconds(5));
            }
        }

        if (initiator_) {
            initiator_->stop();
            initiator_.reset();
        }
    }

    // ----- Данные -----
    QUICKFIX::SessionSettings settings_;
    std::atomic<bool> running_;
    std::thread worker_;

    std::mutex ticks_mutex_;
    std::unordered_map<std::string, MarketTick> ticks_;

    std::unique_ptr<QUICKFIX::ThreadedSocketInitiator> initiator_;

    std::optional<std::chrono::steady_clock::time_point> reconnect_timer_;
    const std::chrono::seconds reconnect_delay_{5s};
};
```

Для краткости мы опустили тонкости управления указателями; вы можете адаптировать код в соответствии с вашей архитектурой.
