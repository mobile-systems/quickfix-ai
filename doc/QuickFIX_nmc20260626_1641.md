## 1. Пример кода

```cpp
// ---------------------------------------------------------------------------
//  get_quotes.cpp  –  простое приложение QuickFIX для получения котировок EURUSD
//                      и их сохранения в PostgreSQL
// ---------------------------------------------------------------------------

#include <iostream>
#include <string>
#include <vector>
#include <chrono>
#include <ctime>
#include <iomanip>
#include <sstream>

#include <quickfix/Application.h>
#include <quickfix/FileLog.h>
#include <quickfix/FileStore.h>
#include <quickfix/Fields.h>
#include <quickfix/MessageCracker.h>
#include <quickfix/Mutex.h>
#include <quickfix/Session.h>
#include <quickfix/SessionID.h>
#include <quickfix/Logger.h>
#include <quickfix/Utility.h>
#include <quickfix/fix44/MarketDataIncrementalRefresh.h>
#include <quickfix/fix44/MarketDataSnapshotSend.h>
#include <quickfix/fix44/Message.h>

// ---------------------------------------------------------------------------
//  1.1  Вспомогательная функция для работы с PostgreSQL (libpq)
// ---------------------------------------------------------------------------

struct PGConnection {
    PGconn *conn;
    PGConnection(const std::string &connStr) {
        conn = PQconnectdb(connStr.c_str());
        if (PQstatus(conn) != CONNECTION_OK) {
            std::cerr << "Подключение к PostgreSQL не удалось: " << PQerrorMessage(conn)
                      << std::endl;
            std::exit(-1);
        }
    }
    ~PGConnection() {
        if (conn) {
            PQfinish(conn);
        }
    }
    bool exec(const std::string &sql) {
        PGresult *res = PQexec(conn, sql.c_str());
        if (PQresultStatus(res) != PGRES_COMMAND_OK) {
            std::cerr << "SQL EXEC ERROR: " << PQerrorMessage(conn) << std::endl;
            PQclear(res);
            return false;
        }
        PQclear(res);
        return true;
    }
};

// ---------------------------------------------------------------------------
//  2.0  Класс приложения, который обрабатывает Market‑Data сообщения
// ---------------------------------------------------------------------------

class QuoteApp : public QUICKFIX::Application, public QUICKFIX::MessageCracker {
public:
    QuoteApp(const std::string &pgConnStr) : pgConn_(pgConnStr) {
        // Создаем таблицу, если ее нет
        pgConn_.exec(
            "CREATE TABLE IF NOT EXISTS eurusd_quotes ("
            "    id SERIAL PRIMARY KEY,"
            "    timestamp TIMESTAMP NOT NULL,"
            "    instrument VARCHAR(10) NOT NULL,"
            "    price NUMERIC(10,5) NOT NULL,"
            "    side CHAR(1) NOT NULL"
            ");");
    }

    // -----------------------------------------------------------------
    //  2.1  Обработчик MarketDataIncrementalRefresh (тип 2)
    // -----------------------------------------------------------------
    void onMessage(const QUICKFIX::MarketDataIncrementalRefresh &msg) override {
        try {
            // Ожидаем, что инструмент EURUSD публикуется в этом сообщении.
            // QuickFix передает список инструментов в поле `NoMDEntries`,
            // а сами записи содержатся в `entries`.
            // В QuickFix 1.19+ вы можете получить записи с помощью:
            //   msg.getMDEntries()
            // Каждая запись – это `MarketDataEntry` (или `QuickFix::FieldNotFound`, если версия старая).

            // Упростим обработку: перебираем все записи, ищем EURUSD.
            // (Для более чистого кода вы бы использовали MessageCracker и переопелили
            //  onMessage для MarketDataIncrementalRefresh.)

            // В QuickFix записи находятся в `msg.getMDEntries()`.
            // `getMDEntries` возвращает `std::vector<QuickFix::MarketDataEntry>`
            // и каждая запись содержит:
            //   - entryType   (буква, например '0' = Бид/Аск)
            //   - instrument  – строка, например "EURUSD"
            //   - price       – строка (относящееся к `Price`)
            //   - currency    – строка
            //   – и т.д.

            // -----------------------------------------------------------------
            //  2.2  Получение цены
            // -----------------------------------------------------------------
            const std::vector<QUICKFIX::MarketDataEntry> &entries = msg.getMDEntries();
            for (const auto &e : entries) {
                // QuickFix может хранить инструмент в поле `Instrument` в записи.
                // (В QuickFix 2.0+ запись содержит поле `instrument` типа std::string)
                // В более старых версиях QuickFix запись неявно содержит "EURUSD"
                // в строке записи; мы просто проверяем строковое представление.
                std::string instrument = e.instrument; // предположим, что есть поле .instrument

                // Приводим цену к double (поле `price` в записи – это строка)
                double price = std::stod(e.price); // e.price – это строка, например "1.1200"

                // Определяем сторону (b = бид, a = оффер)
                char side = (e.entryType == '0') ? 'B' : 'A'; // упрощенная логика

                // -----------------------------------------------------------------
                //  2.3  Сохранение в PostgreSQL
                // -----------------------------------------------------------------
                std::string sql = fmt::format(
                    "INSERT INTO eurusd_quotes (timestamp, instrument, price, side) "
                    "VALUES ('{}', '{}', {}, '{}');",
                    currentTimestamp(),
                    instrument,
                    price,
                    side
                );

                pgConn_.exec(sql);
                std::cout << "Saved: " << instrument << " = " << price << " side=" << side
                          << std::endl;
            }
        } catch (const std::exception &ex) {
            std::cerr << "onMessage exception: " << ex.what() << std::endl;
        }
    }

    // -------------------------------------------------------------------------
    //  2.4  Утилита: форматирование текущей метки времени в виде строки для PostgreSQL
    // -------------------------------------------------------------------------
    std::string currentTimestamp() {
        auto now = std::chrono::system_clock::now();
        auto in_time_t = std::chrono::system_clock::to_time_t(now);
        std::stringstream ss;
        ss << std::put_time(std::gmtime(&in_time_t), "%Y-%m-%d %H:%M:%S");
        return ss.str();
    }

private:
    PGConnection pgConn_;
};

// ---------------------------------------------------------------------------
//  3.0  Основная функция – инициализация сессии QuickFIX и запуск цикла обработки
// ---------------------------------------------------------------------------

int main() {
    // -------------------------------------------------------------------------
    //  3.1  Параметры подключения к FIX (схема acceptor)
    // -------------------------------------------------------------------------
    // В QuickFIX-схеме acceptor мы запускаем слушающий сокет, принимаем сообщения от клиента.
    // Для простоты мы просто создаем сессию с помощью FileStore и FileLog.
    // (Если вам нужна реальная сетевая сессия, вы можете настроить `SocketInitiator`.)

    std::string storePath = "crashes.1";
    std::string logPath = "log.txt";

    try {
        // -------------------------------------------------------------------------
        //  3.2  Создаем хранилище и логгер
        // -------------------------------------------------------------------------
        QUICKFIX::FileStoreFactory storeFactory(storePath);
        QUICKFIX::FileLogFactory logFactory(logPath);

        // -------------------------------------------------------------------------
        //  3.3  Настраиваем параметры сессии (параметры подключения FIX)
        // -------------------------------------------------------------------------
        QUICKFIX::SessionSettings settings;
        settings.set(QUICKFIX::CFG::APPEARANCE, "header");
        settings.set(QUICKFIX::CFG::BEGIN_STRING, "FIX.4.4");
        settings.set(QUICKFIX::CFG::TRANNSENDTIMEOUT, "10");
        settings.set(QUICKFIX::CFG::TRANSSENDTIMEOUT, "10");
        settings.set(QUICKFIX::CFG::TESTREQINTERVAL, "5");

        // -------------------------------------------------------------------------
        //  3.4  Создаем приложение (ваш класс QuoteApp)
        // -------------------------------------------------------------------------
        std::string pgConnStr = "host=localhost dbname=test user=postgres password=secret";
        QuoteApp app(pgConnStr);

        // -------------------------------------------------------------------------
        //  3.5  Создаем инициатор/акцептор – здесь мы используем `SocketAcceptor`,
        //         чтобы слушать сообщения от клиента.
        // -------------------------------------------------------------------------
        QUICKFIX::SocketAcceptor acceptor(app, settings, logFactory);
        // acceptor.start(); // запускает цикл обработки

        // -------------------------------------------------------------------------
        //  3.6  Запуск обработки (простая реализация: цикл с небольшим сном)
        // -------------------------------------------------------------------------
        while (true) {
            acceptor.poll(); // опрос (если реализовано в версии QuickFIX)
            std::this_thread::sleep_for(std::chrono::milliseconds(100));
        }

    } catch (const std::exception &e) {
        std::cerr << "Exception in main: " << e.what() << std::endl;
        return -1;
    }

    return 0;
}
```

### Как это работает (кратко)

| Шаг | Что происходит |
|------|--------------|
| **1** | **QuoteApp** наследуется от `QUICKFIX::Application` и **MessageCracker**. Переопределенный метод `onMessage` принимает `MarketDataIncrementalRefresh` (тип 2) – стандартный FIX-сообщение, содержащий текущие котировки рынка. |
| **2** | В `onMessage` мы извлекаем **Vector** записей MD, ищем **EURUSD** (инструмент обычно указан в поле `instrument` записи), извлекаем цену в виде строки, преобразуем ее в `double`, определяем сторону (бид/аффер) на основе типа записи (`entryType`) и **сохраняем** результат в PostgreSQL. |
| **3** | **Подключение к PostgreSQL** осуществляется с помощью `PGConnection`, обертки над libpq. Функция `exec` просто выполняет запрос `INSERT`. |
| **4** | В **main** мы настраиваем **FileStore** и **FileLog**, создаем экземпляр `QuoteApp` и запускаем **SocketAcceptor**, чтобы приложение могло принимать FIX-сообщения от FIX-клиента (например, от поставщика котировок). |
| **5** | Процесс приема работает в цикле `while(true)`, вызывая `acceptor.poll()` (или `acceptor.run()` в некоторых версиях) для получения входящих сообщений. Каждый обработанный MD-запись записывается в базу данных. |
